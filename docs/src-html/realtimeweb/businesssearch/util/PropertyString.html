001package realtimeweb.businesssearch.util;
002
003import java.util.ArrayList;
004
005/**
006 * This is a convenient class for converting a list of key-value pairs into a
007 * nicely-formatted string.
008 * 
009 * @author acbart
010 * 
011 */
012public class PropertyString {
013        private String name;
014        private ArrayList<String> properties;
015
016        /**
017         * Create a new PropertyString with the given name.
018         * @param name
019         */
020        public PropertyString(String name) {
021                this.name = name;
022                this.properties = new ArrayList<String>();
023        }
024
025        /**
026         * Add a new key-value property pair to the list of properties.
027         * @param property
028         * @param value
029         * @return
030         */
031        public PropertyString add(String property, String value) {
032                this.properties.add(property + "=" + value);
033                return this;
034        }
035
036        /**
037         * Return a nicely-formatted string representation of this PropertyString.
038         */
039        public String toString() {
040                return name + "{" + Util.join(", ", properties) + "}";
041        }
042}
