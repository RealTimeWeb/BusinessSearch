001    package realtimeweb.businesssearch.util;
002    
003    import java.util.ArrayList;
004    import java.util.HashMap;
005    import java.util.List;
006    import java.util.Map;
007    
008    import org.json.simple.parser.ContainerFactory;
009    import org.json.simple.parser.JSONParser;
010    import org.json.simple.parser.ParseException;
011    
012    /**
013     * An internal class that provides a convenient converter for JSON formatted
014     * data. This class will convert JSON maps to HashMaps and JSON lists to
015     * ArrayLists.
016     * 
017     * @author acbart
018     * 
019     */
020    public class JsonConverter {
021            private static ContainerFactory CONTAINER_FACTORY = new ContainerFactory() {
022                    @SuppressWarnings("rawtypes")
023                    public List creatArrayContainer() {
024                            return new ArrayList();
025                    }
026    
027                    @SuppressWarnings("rawtypes")
028                    public Map createObjectContainer() {
029                            return new HashMap();
030                    }
031    
032            };
033    
034            /**
035             * Given a string of JSON data, convert it to a Map.
036             * @param input
037             * @return
038             * @throws ParseException
039             */
040            @SuppressWarnings("unchecked")
041            public static Map<String, Object> convertToMap(String input)
042                            throws ParseException {
043                    return (Map<String, Object>) (new JSONParser()).parse(input,
044                                    CONTAINER_FACTORY);
045            }
046    
047            /**
048             * Given a string of JSON data, convert it to a List.
049             * @param input
050             * @return
051             * @throws ParseException
052             */
053            @SuppressWarnings("unchecked")
054            public static List<Object> convertToList(String input)
055                            throws ParseException {
056                    return (List<Object>) (new JSONParser()).parse(input,
057                                    CONTAINER_FACTORY);
058            }
059    }
