003/**
004 * Indicates an internal server error has occurred. You should probably just try
005 * making the call again, and hope for the best. If it happens more than twice,
006 * contact the API developer!
007 * 
008 * @author acbart
009 * 
010 */
011public class InternalErrorException extends BusinessSearchException {
012
013        private static final long serialVersionUID = 2652326352396426516L;
014
015        /**
016         * Indicates an internal server error has occurred. You should probably just
017         * try making the call again, and hope for the best. If it happens more than
018         * twice, contact the API developer!
019         * 
020         * @param id
021         *            The ID of the exception
022         * @param message
023         *            Some more information about the exception
024         * @param details
025         *            Specific details about the exception
026         */
027        public InternalErrorException(String id, String message, String details) {
028                super(id, message, details);
029        }
030
031}
