006 * data instead.
007 * 
008 * @author acbart
009 * 
010 */

012
013        private static final long serialVersionUID = -5936891507429456619L;
014
015        /**
016         * An exception thrown when a call to the online Yelp service fails,
017         * possibly because of a lost Internet connection. Possibly try using the
018         * local client data instead.
019         * 
020         * @param id
021         *            The ID of the exception
022         * @param message
023         *            Some more information about the exception
024         * @param details
025         *            Specific details about the exception
026         */
027        public DisconnectionException(String id, String message, String fields) {
028                super(id, message, fields);
029        }
030
031}
