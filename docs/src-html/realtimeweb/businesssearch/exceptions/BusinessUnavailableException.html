001package realtimeweb.businesssearch.exceptions;
002
003/**
004 * Indicates that there may not be able to provide details for certain
005 * businesses, for example if they don't have any reviews yet.
006 * 
007 * @author acbart
008 * 
009 */
010public class BusinessUnavailableException extends BusinessSearchException {
011        private static final long serialVersionUID = -7764097544743937114L;
012
013        /**
014         * Indicates that there may not be able to provide details for certain
015         * businesses, for example if they don't have any reviews yet.
016         * 
017         * @param id
018         *            The ID of the exception
019         * @param message
020         *            Some more information about the exception
021         * @param details
022         *            Specific details about the exception
023         */
024        public BusinessUnavailableException(String id, String message,
025                        String details) {
026                super(id, message, details);
027        }
028
029}
