001    package realtimeweb.businesssearch.exceptions;
002    
003    /**
004     * A generic exception thrown when there is an error with the Yelp service. The
005     * ID indicates if it is a local, client side error (e.g. no internet
006     * connection) or if it's a Yelp error. More information about Yelp errors can
007     * be found on the <a
008     * href='http://www.yelp.com/developers/documentation/v2/errors'>Yelp help
009     * page</a>
010     * 
011     * Additional information can also sometimes be found by inspecting the "fields"
012     * property.
013     * 
014     * @author acbart
015     * 
016     */
017    public class BusinessSearchException extends Exception {
018            private static final long serialVersionUID = 1933785601010800054L;
019    
020            /** An alphanumeric string that defines the class of exception **/
021            private String id;
022            /** A string that has more information about the exception. **/
023            private String message;
024            /**
025             * A string that, if present, can give very specific and detailed
026             * information about the exception.
027             **/
028            private String details;
029    
030            /**
031             * @param id
032             *            The ID of the exception
033             * @param message
034             *            Some more information about the exception
035             * @param details
036             *            Specific details about the exception
037             */
038            public BusinessSearchException(String id, String message, String details) {
039                    super(message);
040                    this.id = id;
041                    this.message = message;
042                    this.details = details;
043            }
044    
045            /**
046             * Returns a string representation of the exception (specifically, its ID
047             * and message, not its details).
048             */
049            @Override
050            public String toString() {
051                    return "ERROR " + this.id + ": " + this.message;
052            }
053    
054            /**
055             * Gets the ID of the exception. More information about this ID may be
056             * available on <a
057             * href='http://www.yelp.com/developers/documentation/v2/errors'>the Yelp
058             * error page</a>, unless it is "local", which indicates a client error (in
059             * which case you should check the Message).
060             * 
061             * @return
062             */
063            public String getId() {
064                    return id;
065            }
066    
067            /**
068             * Gets more information about the exception. Usually a single sentence
069             * briefly describing this type of exception.
070             */
071            @Override
072            public String getMessage() {
073                    return message;
074            }
075    
076            /**
077             * Optional. Might contain more detailed information about the specific kind
078             * of exception that occurred.
079             * 
080             * @return
081             */
082            public String getDetails() {
083                    return details;
084            }
085    
086            public static BusinessSearchException generateSpecificException(String id,
087                            String message, String details) {
088                    if (id.equals("INTERNAL_ERROR")) {
089                            return new InternalErrorException(id, message, details);
090                    } else if (id.equals("EXCEEDED_REQS")) {
091                            return new ExceededReqsException(id, message, details);
092                    } else if (id.equals("MISSING_PARAMETER")) {
093                            return new MissingParameterException(id, message, details);
094                    } else if (id.equals("INVALID_PARAMETER")) {
095                            return new InvalidParameterException(id, message, details);
096                    } else if (id.equals("INVALID_SIGNATURE")) {
097                            return new InvalidSignatureException(id, message, details);
098                    } else if (id.equals("INVALID_CREDENTIALS")) {
099                            return new InvalidCredentialsException(id, message, details);
100                    } else if (id.equals("INVALID_OAUTH_CREDENTIALS")) {
101                            return new InvalidOAuthCredentialsException(id, message, details);
102                    } else if (id.equals("INVALID_OAUTH_USER")) {
103                            return new InvalidOAuthUserException(id, message, details);
104                    } else if (id.equals("ACCOUNT_UNCONFIRMED")) {
105                            return new AccountUnconfirmedException(id, message, details);
106                    } else if (id.equals("PASSWORD_TOO_LONG")) {
107                            return new PasswordTooLongException(id, message, details);
108                    } else if (id.equals("UNAVAILABLE_FOR_LOCATION")) {
109                            return new UnavailableForLocationException(id, message, details);
110                    } else if (id.equals("AREA_TOO_LARGE")) {
111                            return new AreaTooLargeException(id, message, details);
112                    } else if (id.equals("MULTIPLE_LOCATIONS")) {
113                            return new MultipleLocationsException(id, message, details);
114                    } else if (id.equals("BUSINESS_UNAVAILABLE")) {
115                            return new BusinessUnavailableException(id, message, details);
116                    } else {
117                            return new BusinessSearchException(id, message, details);
118                    }
119            }
120    }
