001    package realtimeweb.businesssearch.json;
002    
003    import realtimeweb.businesssearch.exceptions.BusinessSearchException;
004    
005    /**
006     * A listener for handling data received about a specific Business. On success,
007     * a string of JSON containing all the information about the Business will be
008     * passed to the businessDataCompleted method, which must be overridden in any
009     * implementing classes. On failure, an exception is passed to the
010     * businessDataFailed method, which must also be overridden in any implementing
011     * class.<br>
012     * <br>
013     * 
014     * Details about the JSON string returned can be found in the <a
015     * href='http://www.yelp.com/developers/documentation/v2/business'>Yelp API
016     * documentation.</a>
017     * 
018     * @author acbart
019     * 
020     */
021    public interface JsonBusinessDataListener {
022            /**
023             * The method that should be overridden to handle the JSON-formatted
024             * business data. For reference on the data being returned, consult the <a
025             * href='http://www.yelp.com/developers/documentation/v2/business'>Yelp API
026             * documentation</a>.
027             * 
028             * @param businessData
029             */
030            public abstract void businessDataCompleted(String businessData);
031    
032            /**
033             * The method that should be overridden to handle the Exception that occurs
034             * while getting the Business Data.
035             * 
036             * @param exception
037             */
038            public abstract void businessDataFailed(BusinessSearchException exception);
039    }
