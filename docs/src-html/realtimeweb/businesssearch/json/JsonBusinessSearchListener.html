001    package realtimeweb.businesssearch.json;
002    
003    import realtimeweb.businesssearch.exceptions.BusinessSearchException;
004    
005    /**
006     * A listener for handling data received about a list of businesses. On success,
007     * a string of JSON containing all the information about the List of Businesses
008     * will be passed to the businessSearchCompleted method, which must be
009     * overridden in any implementing classes. On failure, an exception is passed to
010     * the businessSearchFailed method, which must also be overridden in any
011     * implementing class.<br>
012     * <br>
013     * 
014     * Details about the JSON string returned can be found in the <a
015     * href='http://www.yelp.com/developers/documentation/v2/search_api'>Yelp API
016     * documentation.</a>
017     * 
018     * @author acbart
019     * 
020     */
021    public interface JsonBusinessSearchListener {
022            /**
023             * The method that should be overridden to handle the JSON-formatted list of
024             * businesses data. For reference on the data being returned, consult the <a
025             * href='http://www.yelp.com/developers/documentation/v2/search_api'>Yelp
026             * API documentation.</a>
027             * 
028             * @param businessData
029             */
030            public abstract void businessSearchCompleted(String searchResponse);
031    
032            /**
033             * The method that should be overridden to handle an exception that occurred
034             * while getting the JSON response of the query.
035             * 
036             * @param exception
037             */
038            public abstract void businessSearchFailed(BusinessSearchException exception);
039    }
