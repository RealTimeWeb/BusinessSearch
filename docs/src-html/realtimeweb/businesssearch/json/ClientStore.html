001    package realtimeweb.businesssearch.json;
002    
003    import java.io.BufferedReader;
004    import java.io.BufferedWriter;
005    import java.io.FileWriter;
006    import java.io.IOException;
007    import java.io.InputStreamReader;
008    import java.util.HashMap;
009    import java.util.Map.Entry;
010    
011    import org.json.simple.JSONValue;
012    import org.json.simple.parser.ParseException;
013    
014    import realtimeweb.businesssearch.util.JsonConverter;
015    
016    /**
017     * This method is intended for internal use only.
018     * 
019     * This is the class that is used to store internal results.
020     * 
021     * @author acbart
022     * 
023     */
024    class ClientStore {
025    
026            private HashMap<String, String> jsonData;
027    
028            /**
029             * Create a client store from a specified file.
030             * 
031             * @param source
032             */
033            public ClientStore(String source, boolean recording) {
034                    jsonData = new HashMap<String, String>();
035                    if (!recording) {
036                            try {
037                                    this.load(source);
038                            } catch (IOException e) {
039                                    System.err
040                                                    .println("Couldn't find client-side datastore! Your Jar might be corrupt.");
041                                    e.printStackTrace();
042                            }
043                    }
044            }
045    
046            /**
047             * Look up a given query in the client store. If it is not found, returns
048             * the empty string.
049             * 
050             * @param key
051             * @return
052             */
053            public String getData(String key) {
054                    if (jsonData.containsKey(key)) {
055                            return jsonData.get(key);
056                    } else {
057                            return "";
058                    }
059            }
060    
061            public void putData(String key, String value) {
062                    jsonData.put(key, value);
063            }
064    
065            /**
066             * Create a client store based on the default store.
067             */
068            public ClientStore(boolean recording) {
069                    this("../cache.json", recording);
070            }
071    
072            /**
073             * Load the client store from a file.
074             * 
075             * @param source
076             * @throws IOException
077             */
078            public void load(String source) throws IOException {
079                    InputStreamReader is = new InputStreamReader(getClass()
080                                    .getResourceAsStream(source));
081                    BufferedReader clientData = new BufferedReader(is);
082                    String line;
083                    while ((line = clientData.readLine()) != null) {
084                            try {
085                                    HashMap<String, Object> hm = (HashMap<String, Object>) JsonConverter
086                                                    .convertToMap(line);
087                                    for (Entry<String, Object> es : hm.entrySet()) {
088                                            jsonData.put(es.getKey(), es.getValue().toString());
089                                    }
090                            } catch (ParseException e) {
091                                    System.err
092                                                    .println("Client-Side Data couldn't be parsed! Your jar might be corrupt.");
093                                    e.printStackTrace();
094                            }
095                    }
096                    clientData.close();
097            }
098    
099            public synchronized void save(String source, boolean append)
100                            throws IOException {
101                    FileWriter fw = new FileWriter(source, append);
102                    BufferedWriter clientFile = new BufferedWriter(fw);
103                    JSONValue.writeJSONString(jsonData, clientFile);
104                    clientFile.close();
105            }
106    }
