001package realtimeweb.businesssearch.json;
002
003import java.io.BufferedReader;
004import java.io.BufferedWriter;
005import java.io.FileWriter;
006import java.io.IOException;
007import java.io.InputStreamReader;
008import java.util.HashMap;
009import java.util.Map.Entry;
010
011import org.json.simple.JSONValue;
012import org.json.simple.parser.ParseException;
013
014import realtimeweb.businesssearch.util.JsonConverter;
015
016/**
017 * This method is intended for internal use only.
018 * 
019 * This is the class that is used to store internal results. 
020 * @author acbart
021 *
022 */
023class ClientStore {
024
025        private HashMap<String, String> jsonData;
026        
027        /**
028         * Create a client store from a specified file.
029         * @param source
030         */
031        public ClientStore(String source, boolean recording) {
032                jsonData = new HashMap<String, String>();
033                if (!recording) {
034                        try {
035                                this.load(source);
036                        } catch (IOException e) {
037                                System.err.println("Couldn't find client-side datastore! Your Jar might be corrupt.");
038                                e.printStackTrace();
039                        }
040                }
041        }
042        
043        /**
044         * Look up a given query in the client store. If it is not found, returns the empty string.
045         * @param key
046         * @return
047         */
048        public String getData(String key) {
049                if (jsonData.containsKey(key)) {
050                        return jsonData.get(key);
051                } else {
052                        return "";
053                }
054        }
055        
056        public void putData(String key, String value) {
057                jsonData.put(key, value);
058        }
059
060        /**
061         * Create a client store based on the default store.
062         */
063        public ClientStore(boolean recording) {
064                this("../cache.json", recording);
065        }
066
067        /**
068         * Load the client store from a file.
069         * @param source
070         * @throws IOException
071         */
072        public void load(String source) throws IOException {
073                InputStreamReader is = new InputStreamReader(getClass().getResourceAsStream(source));
074                BufferedReader clientData = new BufferedReader(is);
075                String line;
076                while ((line = clientData.readLine()) != null) {
077                        try {
078                                HashMap<String, Object> hm = (HashMap<String, Object>) JsonConverter.convertToMap(line);
079                                for (Entry<String, Object> es : hm.entrySet()) {
080                                        jsonData.put(es.getKey(), es.getValue().toString());
081                                }
082                        } catch (ParseException e) {
083                                System.err.println("Client-Side Data couldn't be parsed! Your jar might be corrupt.");
084                                e.printStackTrace();
085                        }
086                }
087                clientData.close();
088        }
089        
090        public synchronized void save(String source, boolean append) throws IOException {
091                FileWriter fw = new FileWriter(source, append);
092                BufferedWriter clientFile = new BufferedWriter(fw);
093                JSONValue.writeJSONString(jsonData, clientFile);
094                clientFile.close();
095        }
096}
