022    /**
023     * The JsonBusinessSearch class is used to connect to the Yelp service and
024     * retrieve data JSON formatted data. All data is returned asynchronously - your
025     * program will continue to run while the data is being fetched, and the result
026     * will be sent to a callback function that you specify.<br>
027     * <br>
028     * 
029     * There will only be one instance of the StructuredBusinessSearch, because it
030     * follows the Singleton pattern. In order to use it, first get the instance of
031     * the class.
032     * 
033     * <pre>
034     * StructuredBusinessSearch sbs = StructuredBusinessSearch.getInstance();
035     * </pre>
036     * 
037     * You can use the service offline immediately, or connect to the online
038     * service:
039     * 
040     * <pre>
041     * sbs.connect(&quot;key&quot;, &quot;secret&quot;, &quot;token&quot;, &quot;secrettoken&quot;);
042     * </pre>
043     * 
044     * If you need to disconnect, it is simple:
045     * 
046     * <pre>
047     * sbs.disconnect();
048     * </pre>
049     * 
050     * @author acbart
051     * 
052     */
053    public class JsonBusinessSearch implements AbstractBusinessSearch {

084                                    if (instance == null) {

088                    }

112            /**
113             * A protected constructor to defeat instantation of this singleton.
114             */
115            protected JsonBusinessSearch(boolean recording) {
116                    disconnect();
117                    setStoring(recording);
118                    this.service = null;
119                    this.accessToken = null;
120                    this.clientStore = new ClientStore(recording);
121                    BusinessCategory.getCategories();
122            }

124            /**
125             * Establishes a connection to the online Business Search service.
126             * dataservice. This does not require an internet connection.<br><br>
127             * 
128             * Requires registration information from Yelp. To get your key go to
129             * <a href='http://www.yelp.com/developers'>http://www.yelp.com/developers</a>
130             * 
131             * @param consumerKey
132             * @param consumerSecret
133             * @param token
134             * @param tokenSecret
135             */
136            public void connect(String consumerKey, String consumerSecret,
137                            String token, String tokenSecret) {
138                    this.local = false;
139                    this.service = new ServiceBuilder()
140                                    .provider(ServiceAuthentication.class).apiKey(consumerKey)
141                                    .apiSecret(consumerSecret).build();
142                    this.accessToken = new Token(token, tokenSecret);
143            }
144    
145            /**
146             * Establishes that data should be retrieved locally instead of from the
147             * online service. This is the default behavior for the BusinessSearch.<br>
148             * <br>
149             * 
150             * If data is being retrieved locally, you must be sure that your parameters
151             * match locally stored data. Otherwise, you will get nothing in return.
152             * 
153             * @param local
154             */
155            public void disconnect() {
156                    this.local = true;
157            }
158    
159            /**
160             * The function is used to create a request to the server. By converting a
161             * request to a string, you can get an accurate and storeable version of it.
162             * 
163             * @param destination
164             * @param hashMap
165             * @return
166             */
167            OAuthRequest makeRequest(String destination, HashMap<String, String> hashMap) {
168                    OAuthRequest request = new OAuthRequest(Verb.GET,
169                                    "http://api.yelp.com/v2/" + destination);
170                    for (Entry<String, String> parameter : hashMap.entrySet()) {
171                            String key = parameter.getKey();
172                            String value = parameter.getValue();
173                            request.addQuerystringParameter(key, value);
174                    }
175                    if (!this.local) {
176                            this.service.signRequest(this.accessToken, request);
177                    }
178                    return request;
179            }
180    
181            /**
182             * This function is used to send a request to the server.
183             * 
184             * @param request
185             * @return
186             */
187            String sendRequest(OAuthRequest request) throws OAuthException {
188                    String hashedRequest = hashRequest(request);
189                    if (local) {
190                            return clientStore.getData(hashedRequest);
191                    } else {
192                            String response = request.send().getBody();
193                            if (storing) {
194                                    clientStore.putData(hashedRequest, response);
195                            }
196                            return response;
197                    }
198            }
199    
200            /**
201             * Convert a Request object to an injective hash.<br><br>
202             * 
203             * An injective hash has no collisions. Hopefully that can't occur with
204             * this.
205             * 
206             * @param request
207             * @return
208             */
209            @SuppressWarnings({ "unchecked", "rawtypes" })
210            private static String hashRequest(OAuthRequest request) {
211                    String url = request.getUrl();
212                    String queryStrings = (new TreeMap(request.getQueryStringParams()))
213                                    .toString();
214                    String bodyParams = (new TreeMap(request.getBodyParams())).toString();
215                    return url + "%" + queryStrings + "%" + bodyParams;
216            }
217    
218            /**
219             * A method that will convert a Yelp error to a Java exception, if there is
220             * one.
221             * 
222             * @param response
223             * @return
224             */
225            BusinessSearchException checkErrors(String response) {
226                    if (response.isEmpty()) {
227                            if (local) {
228                                    return null;
229                            } else {
230                                    return new BusinessSearchException(
231                                                    "Local",
232                                                    "Could not connect to the online Yelp service. Try using a local connection instead.",
233                                                    "");
234                            }
235                    }
236                    try {
237                            HashMap<String, Object> r = (HashMap<String, Object>) JsonConverter
238                                            .convertToMap(response);
239                            if (r.containsKey("error")) {
240                                    @SuppressWarnings("unchecked")
241                                    HashMap<String, Object> error = (HashMap<String, Object>) r
242                                                    .get("error");
243                                    return new BusinessSearchException((String) error.get("id"),
244                                                    (String) error.get("text"), (String) error.get("field"));
245                            }
246                    } catch (ParseException e) {
247                            return new BusinessSearchException(
248                                            "Local",
249                                            "The response from the Yelp Service was not valid JSON. Check the fields for more information on what was returned.",
250                                            response);
251                    }
252                    return null;
253            }

255            /**
256             * Returns data about a business given its businessId (which is a sequence
257             * of characters, symbols, and numbers, e.g. "I_7cQmHxx6LokP9yr7Fx-w").
258             * 
259             * @param businessId
260             *            A string identifying the business.
261             * @param listener
262             *            a callback function that will consume the data (or error)
263             *            about the business.
264             */
265            public void getBusinessData(final String businessId,
266                            final JsonBusinessDataListener listener) {
267                    final JsonBusinessSearch _this = this;
268                    Thread thread = new Thread() {
269                            public void run() {
270                                    OAuthRequest request = _this.makeRequest("business/"
271                                                    + businessId, new HashMap<String, String>());
272                                    String response = _this.sendRequest(request);
273                                    BusinessSearchException error = _this.checkErrors(response);
274                                    if (error == null) {
275                                            listener.businessDataCompleted(response);
276                                    } else {
277                                            listener.businessDataFailed(error);
278                                    }
279                            }
280                    };
281                    thread.start();
282            }
283    
284            /**
285             * Queries the Yelp service for businesses that match the given
286             * BusinessQuery. Returns the data (or error) through the listener object.
287             * 
288             * @param query
289             *            A BusinessQuery object
290             * @param listener
291             *            A listener object that should contain methods for handling the
292             *            data and, optionally, any errors.
293             */
294            public void searchBusinesses(final BusinessQuery query,
295                            final JsonBusinessSearchListener listener) {
296                    final JsonBusinessSearch _this = this;
297                    Thread thread = new Thread() {
298                            public void run() {
299                                    OAuthRequest request = _this.makeRequest("search",
300                                                    query.getFilters());
301                                    String response = _this.sendRequest(request);
302                                    BusinessSearchException error = _this.checkErrors(response);
303                                    if (error == null) {
304                                            listener.businessSearchCompleted(response);
305                                    } else {
306                                            listener.businessSearchFailed(error);
307                                    }
308                            }
309                    };
310                    thread.start();
311            }
