018/**
019 * The StructuredBusinessSearch class is used to connect to the Yelp service and
020 * retrieve data as a Map. All data is returned asynchronously - your program
021 * will continue to run while the data is being fetched, and the result will be
022 * sent to a callback function that you specify.<br>
023 * <br>
024 * 
025 * There will only be one instance of the StructuredBusinessSearch, because it
026 * follows the Singleton pattern. In order to use it, first get the instance of
027 * the class.
028 * 
029 * <pre>
030 * StructuredBusinessSearch sbs = StructuredBusinessSearch.getInstance();
031 * </pre>
032 * 
033 * Then you can either immediately use the service offline, or connect to the online service:
034 * 
035 * <pre>
036 * sbs.connect(&quot;key&quot;, &quot;secret&quot;, &quot;token&quot;, &quot;secrettoken&quot;);
037 * </pre>
038 * 
039 * To disconnect from the online service, simply:
040 * 
041 * <pre>
042 * sbs.disconnectLocal();
043 * </pre>
044 * 
045 * @author acbart
046 * 
047 */
048public class StructuredBusinessSearch implements AbstractBusinessSearch {
049        private static StructuredBusinessSearch instance;
050
051        /**
052         * Retrieves the singleton instance of the JsonBusinessSearch.
053         * 
054         * @return
055         */
056        public static StructuredBusinessSearch getInstance() {
057                if (instance == null) {
058                        synchronized (StructuredBusinessSearch.class) {
059                                if (instance == null) {
060                                        instance = new StructuredBusinessSearch();
061                                }
062                        }
063                }
064                return instance;
065        }
066
067        /**
068         * Protected constructor to guard against instantiation.
069         */
070        protected StructuredBusinessSearch() {
071                jsonBusinessSearch = JsonBusinessSearch.getInstance();
072        }
073
074        private JsonBusinessSearch jsonBusinessSearch;
075
076        /**
077         * Returns data about a business given its businessId (which is a sequence
078         * of characters, symbols, and numbers, e.g. "I_7cQmHxx6LokP9yr7Fx-w").
079         * 
080         * @param businessId
081         *            A string identifying the business.
082         * @param listener
083         *            a callback function that will consume the data (or error)
084         *            about the business.
085         */
086        public void getBusinessData(String businessId,
087                        final StructuredBusinessDataListener listener) {
088                this.jsonBusinessSearch.getBusinessData(businessId,
089                                new JsonBusinessDataListener() {
090
091                                        @Override
092                                        public void businessDataCompleted(String business) {
093                                                HashMap<String, Object> structuredBusiness = null;
094                                                if (business == "") {
095                                                        listener.businessDataCompleted(StructuredBusinessSearch.getEmptyBusiness());
096                                                        return;
097                                                }
098                                                try {
099                                                        structuredBusiness = (HashMap<String, Object>) JsonConverter
100                                                                        .convertToMap(business);
101                                                } catch (ParseException e) {
102                                                        listener.businessDataFailed(new DataUnreadableException(
103                                                                        "INVALID_DATA",
104                                                                        "The JSON data received from the server couldn't be parsed.",
105                                                                        e.toString()));
106                                                }
107                                                listener.businessDataCompleted(structuredBusiness);
108                                        }
109
110                                        @Override
111                                        public void businessDataFailed(BusinessSearchException exception) {
112                                                listener.businessDataFailed(exception);
113                                        }
114
115                                });
116        }
117
118        @SuppressWarnings("rawtypes")
119        protected static Map<String, Object> getEmptyBusiness() {
120                HashMap<String, Object> empty = new HashMap<String, Object>();
121                empty.put("categories", new ArrayList());
122                empty.put("deals", new ArrayList());
123                empty.put("gift_certificates", new ArrayList());
124                empty.put("reviews", new ArrayList());
125                empty.put("id", "");
126                empty.put("display_phone", "");
127                empty.put("image_url", "");
128                empty.put("is_claimed", false);
129                empty.put("is_closed", false);
130                empty.put("mobile_url", "");
131                empty.put("name", "");
132                empty.put("phone", "");
133                empty.put("rating", 1.0);
134                empty.put("rating_img_url", "");
135                empty.put("rating_img_url_large", "");
136                empty.put("rating_img_url_small", "");
137                empty.put("review_count", 0);
138                empty.put("snippet_image_url", "");
139                empty.put("snippet_text", "");
140                empty.put("url", "");
141                
142                HashMap<String, Object> location = new HashMap<String, Object>();
143                location.put("address", new ArrayList());
144                location.put("display_address", new ArrayList());
145                location.put("neighborhoods", new ArrayList());
146                location.put("city", "");
147                location.put("country_code", "");
148                location.put("cross_streets", "");
149                location.put("geo_accuracy", 4);
150                location.put("postal_code", "");
151                location.put("state_code", "");
152                HashMap<String, Object> coordinate= new HashMap<String, Object>();
153                coordinate.put("latitude", 0.0);
154                coordinate.put("longitude", 0.0);
155                location.put("coordinate", coordinate);
156                
157                empty.put("location", location);
158                
159                return empty;
160        }
161
162        /**
163         * Queries the Yelp service for businesses that match the given
164         * BusinessQuery. Returns the data (or error) through the listener object.
165         * 
166         * @param query
167         *            A BusinessQuery object
168         * @param listener
169         *            A listener object that should contain methods for handling the
170         *            data and, optionally, any errors.
171         */
172        public void searchBusinesses(BusinessQuery query,
173                        final StructuredBusinessSearchListener listener) {
174                this.jsonBusinessSearch.searchBusinesses(query,
175                                new JsonBusinessSearchListener() {
176
177                                        @Override
178                                        public void businessSearchCompleted(String response) {
179                                                HashMap<String, Object> structuredResponse = null;
180                                                if (response == "") {
181                                                        listener.businessSearchCompleted(StructuredBusinessSearch.getEmptySearchReponse());
182                                                        return;
183                                                }
184                                                try {
185                                                        structuredResponse = (HashMap<String, Object>) JsonConverter
186                                                                        .convertToMap(response);
187                                                } catch (ParseException e) {
188                                                        listener.businessSearchFailed(new DataUnreadableException(
189                                                                        "INVALID_DATA",
190                                                                        "The JSON data received from the server couldn't be parsed.",
191                                                                        e.toString()));
192                                                }
193                                                listener.businessSearchCompleted(structuredResponse);
194                                        }
195
196                                        public void businessSearchFailed(BusinessSearchException exception) {
197                                                listener.businessSearchFailed(exception);
198                                        }
199
200                                });
201        }
202
203        protected static Map<String, Object> getEmptySearchReponse() {
204                HashMap<String, Object> empty = new HashMap<String, Object>();
205                empty.put("businesses", new ArrayList<String>());
206                HashMap<String, Object> region = new HashMap<String, Object>();
207                HashMap<String, Object> center= new HashMap<String, Object>();
208                center.put("latitude", 0.0);
209                center.put("longitude", 0.0);
210                HashMap<String, Object> span= new HashMap<String, Object>();
211                span.put("latitude_delta", 0.0);
212                span.put("longitude_delta", 0.0);
213                region.put("center", center);
214                region.put("span", span);
215                empty.put("region", region);
216                empty.put("total", 0);
217                return empty;
218        }

220        /**
221         * Establishes a connection to the online Business Search service.
222         * dataservice. This requires an internet connection.<br><br>
223         * 
224         * Requires registration information from Yelp. To get your key go to
225         * <a href='http://www.yelp.com/developers'>http://www.yelp.com/developers</a>
226         * 
227         * @param consumerKey
228         * @param consumerSecret
229         * @param token
230         * @param tokenSecret
231         */
232        public void connect(String consumerKey, String consumerSecret,
233                        String token, String tokenSecret) {
234                this.jsonBusinessSearch.connect(consumerKey, consumerSecret, token,
235                                tokenSecret);
236        }
237
238        /**
239         * Establishes that Business Search data should be retrieved locally. This
240         * does not require an internet connection.<br><br>
241         * 
242         * If data is being retrieved locally, you must be sure that your parameters
243         * match locally stored data. Otherwise, you will get nothing in return.
244         * 
245         * @param local
246         */
247        public void disconnect() {
248                this.jsonBusinessSearch.disconnect();
249        }
