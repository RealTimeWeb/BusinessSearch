001package realtimeweb.businesssearch.structured;
002
003import java.util.Map;
004
005import realtimeweb.businesssearch.exceptions.BusinessSearchException;
006
007/**
008 * A listener for handling data received about a list of businesses. On success,
009 * a Map containing all the information about the List of Businesses will be
010 * passed to the businessSearchCompleted method, which must be overridden in any
011 * implementing classes. On failure, an exception is passed to the
012 * businessSearchFailed method, which must also be overridden in any
013 * implementing class.<br>
014 * <br>
015 * 
016 * Details about the fields of the Map returned can be found in the Yelp API
017 * documentation: http://www.yelp.com/developers/documentation/v2/search_api
018 * 
019 * @author acbart
020 * 
021 */
022public interface StructuredBusinessSearchListener {
023        /**
024         * The method that should be overridden to handle the Map of the list of
025         * businesses. For reference on the data being returned, consult the Yelp
026         * API documentation:
027         * http://www.yelp.com/developers/documentation/v2/search_api
028         * 
029         * @param businessData
030         */
031        public abstract void businessSearchCompleted(
032                        Map<String, Object> searchResponse);
033
034        /**
035         * The method that should be overridden to handle an exception that occurred
036         * while getting the search response.
037         * 
038         * @param exception
039         */
040        public abstract void businessSearchFailed(BusinessSearchException exception);
041}
