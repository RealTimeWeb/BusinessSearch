001    package realtimeweb.businesssearch.structured;
002    
003    import java.util.Map;
004    
005    import realtimeweb.businesssearch.exceptions.BusinessSearchException;
006    
007    /**
008     * A listener for handling data received about a list of businesses. On success,
009     * a Map containing all the information about the List of Businesses will be
010     * passed to the businessSearchCompleted method, which must be overridden in any
011     * implementing classes. On failure, an exception is passed to the
012     * businessSearchFailed method, which must also be overridden in any
013     * implementing class.<br>
014     * <br>
015     * 
016     * Details about the fields of the Map returned can be found in the <a
017     * href='http://www.yelp.com/developers/documentation/v2/search_api'>Yelp API
018     * documentation.</a>
019     * 
020     * @author acbart
021     * 
022     */
023    public interface StructuredBusinessSearchListener {
024            /**
025             * The method that should be overridden to handle the Map of the list of
026             * businesses. For reference on the data being returned, consult the <a
027             * href='http://www.yelp.com/developers/documentation/v2/search_api'>Yelp
028             * API documentation.</a>
029             * 
030             * @param businessData
031             */
032            public abstract void businessSearchCompleted(
033                            Map<String, Object> searchResponse);
034    
035            /**
036             * The method that should be overridden to handle an exception that occurred
037             * while getting the search response.
038             * 
039             * @param exception
040             */
041            public abstract void businessSearchFailed(BusinessSearchException exception);
042    }
