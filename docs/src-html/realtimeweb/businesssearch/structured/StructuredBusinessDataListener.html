001package realtimeweb.businesssearch.structured;
002
003import java.util.Map;
004
005import realtimeweb.businesssearch.exceptions.BusinessSearchException;
006
007/**
008 * A listener for handling data received about a specific Business. On success,
009 * a Map containing all the information about the Business will be passed to the
010 * businessDataCompleted method, which must be overridden in any implementing
011 * classes. On failure, an exception is passed to the businessDataFailed method,
012 * which must also be overridden in any implementing class.<br>
013 * <br>
014 * 
015 * Details about the fields of the Map returned can be found in the <a
016 * href='http://www.yelp.com/developers/documentation/v2/business'>Yelp API
017 * documentation.</a>
018 * 
019 * @author acbart
020 * 
021 */
022public interface StructuredBusinessDataListener {
023        /**
024         * The method that should be overridden to handle the Map of business data.
025         * For reference on the data being returned, consult the <a
026         * href='http://www.yelp.com/developers/documentation/v2/business'>Yelp API
027         * documentation.</a>
028         * 
029         * @param businessData
030         */
031        public abstract void businessDataCompleted(Map<String, Object> business);
032
033        /**
034         * The method that should be overridden to handle an exception that occurred
035         * while getting the business data.
036         * 
037         * @param exception
038         */
039        public abstract void businessDataFailed(BusinessSearchException exception);
040}
