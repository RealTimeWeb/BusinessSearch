008    /**
009     * A query object used to make a search for businesses. A BusinessQuery is
010     * built-up by chaining together filters.<br>
011     * For example:<br>
012     * <br>
013     * <code>BusinessQuery myQuery = BusinessQuery("Blacksburg, VA");<br>
014     * myQuery.limitRadius(100).filterCategories("restaurants").sortBy(SortMode.DISTANCE);</code>
015     * <br>
016     * <br>
017     * ... will create a query that will search for restaurants within 100 meters of
018     * Blacksburg, and sort the results by Distance.
019     * 
020     * If you wish to remove a filter after it's been added, you can do so easily: <br>
021     * <br>
022     * <code>myQuery.removeRadiusLimit().removeCategoriesFilter();</code><br>
023     * <br>
024     * ... will change the query to search all around Blacksburg for any businesses.
025     * and just sort the results.
026     * 
027     * @author acbart
028     * 
029     */
030    public class BusinessQuery {

180            /**
181             * Filters the results by categories, returning only results that fall into
182             * the given categories. BusinessCategories can be found via the
183             * {@link BusinessCategory#getCategories getCategories} method. An arbitrary
184             * number of categories can be included. You can also search via the
185             * category's alias by passing in Strings, but you cannot mix
186             * BusinessCategories and Strings.
187             * 
188             * @param categories
189             * @return This BusinessQuery instance, allowing you to chain together other
190             *         filters.
191             */
192            public BusinessQuery filterCategories(BusinessCategory... categories) {
193                    this.categories = categories.clone();
194                    this.hasCategories = true;
195                    return this;
196            }
197    
198            /**
199             * Filters the results by categories given by their alias (e.g. General
200             * Dentistry would be "generaldentistry") - so only results that businesses
201             * that are in the category will be returned. An arbitrary number of aliases
202             * can be given. A complete list of all the categories and their aliases is
203             * at <a
204             * href="http://www.yelp.com/developers/documentation/all_category_list"
205             * >http://www.yelp.com/developers/documentation/all_category_list</a>.
206             * 
207             * @param categories
208             * @return This BusinessQuery instance, allowing you to chain together other
209             *         filters.
210             */
211            public BusinessQuery filterCategories(String... categories) {
212                    this.categories = new BusinessCategory[categories.length];
213                    for (int i = 0; i < categories.length; i++) {
214                            this.categories[i] = BusinessCategory.getFromAlias(categories[i],
215                                            "");
216                    }
217                    this.hasCategories = true;
218                    return this;
219            }
220    
221            /**
222             * Remove any categories that are being filtered.
223             * 
224             * @return This BusinessQuery instance, allowing you to chain together other
225             *         filters.
226             */
227            public BusinessQuery removeCategoriesFilter() {
228                    this.hasCategories = false;
229                    return this;
230            }
231    
232            /**
233             * Remove a list of categories from the filter.
234             * 
235             * @param categories
236             * @return This BusinessQuery instance, allowing you to chain together other
237             *         filters.
238             */
239            public BusinessQuery removeCategoriesFilter(BusinessCategory... categories) {
240                    List<BusinessCategory> removingCategories = Arrays.asList(categories);
241                    List<BusinessCategory> entireCategories = Arrays
242                                    .asList(this.categories);
243                    entireCategories.removeAll(removingCategories);
244                    this.categories = entireCategories
245                                    .toArray(new BusinessCategory[entireCategories.size()]);
246                    if (entireCategories.isEmpty()) {
247                            this.hasCategories = false;
248                    }
249                    return this;
250            }
251    
252            /**
253             * Remove a list of categories from the filter, given as alias strings.
254             * 
255             * @param categories
256             * @return This BusinessQuery instance, allowing you to chain together other
257             *         filters.
258             */
259            public BusinessQuery removeCategoriesFilter(String... categories) {
260                    List<BusinessCategory> removingCategories = new ArrayList<BusinessCategory>();
261                    for (int i = 0; i < categories.length; i++) {
262                            removingCategories.add(BusinessCategory.getFromAlias(categories[i],
263                                            ""));
264                    }
265                    List<BusinessCategory> entireCategories = Arrays
266                                    .asList(this.categories);
267                    entireCategories.removeAll(removingCategories);
268                    this.categories = entireCategories
269                                    .toArray(new BusinessCategory[entireCategories.size()]);
270                    if (entireCategories.isEmpty()) {
271                            this.hasCategories = false;
272                    }
273                    return this;
274            }

416            /**
417             * This method is intended for internal use only.<Br>
418             * <br>
419             * 
420             * Converts the internal list of filters into a dictionary.
421             * 
422             * @return
423             */
424            public HashMap<String, String> getFilters() {
425                    HashMap<String, String> filters = new HashMap<String, String>();
426    
427                    filters.putAll(this.location.getFilter());
428    
429                    if (this.hasTerm) {
430                            filters.put("term", this.term);
431                    }
432    
433                    if (this.hasLimit) {
434                            filters.put("limit", "" + this.limit);
435                    }
436    
437                    if (this.hasOffset) {
438                            filters.put("offset", "" + this.offset);
439                    }
440    
441                    if (this.hasCategories && this.categories.length >= 1) {
442                            StringBuilder csl = new StringBuilder();
443                            csl.append(this.categories[0]);
444                            if (this.categories.length > 1) {
445                                    for (int i = 1; i < this.categories.length; i++) {
446                                            csl.append("," + this.categories[i]);
447                                    }
448                            }
449                            filters.put("category_filter", csl.toString());
450                    }
451    
452                    if (this.hasRadius) {
453                            filters.put("radius_filter", "" + this.radius);
454                    }
455    
456                    if (this.hasDeals) {
457                            filters.put("deals_filter", (this.deals ? "1" : "0"));
458                    }
459    
460                    if (this.hasSortMode) {
461                            filters.put("sort", "" + this.sortMode.ordinal());
462                    }
463    
464                    return filters;
465            }
