008/**
009 * A query object used to make a search for businesses. A BusinessQuery is
010 * built-up by chaining together filters.<br>
011 * For example:<br>
012 * <br>
013 * <code>BusinessQuery myQuery = BusinessQuery("Blacksburg, VA");<br>
014 * myQuery.limitRadius(100).filterCategories("restaurants").sortBy(SortMode.DISTANCE);</code>
015 * <br>
016 * <br>
017 * ... will create a query that will search for restaurants within 100 meters of
018 * Blacksburg, and sort the results by Distance.
019 * 
020 * If you wish to remove a filter after it's been added, you can do so easily: <br>
021 * <br>
022 * <code>myQuery.removeRadiusLimit().removeCategoriesFilter();</code><br>
023 * <br>
024 * ... will change the query to search all around Blacksburg for any businesses.
025 * and just sort the results.
026 * 
027 * @author acbart
028 * 
029 */
030public class BusinessQuery {
031

033        private boolean hasRadius;

040        private Location location;

042        private boolean hasCategories;
043        private boolean hasDeals;
044        private boolean hasSortMode;

173        /**
174         * Filters the results by categories, returning only results that fall into
175         * the given categories. BusinessCategories can be found via the
176         * {@link BusinessCategory#getCategories() getCategories} method. An
177         * arbitrary number of categories can be included. You can also search via
178         * the category's alias by passing in Strings, but you cannot mix
179         * BusinessCategories and Strings.
180         * 
181         * @param categories
182         * @return This BusinessQuery instance, allowing you to chain together other
183         *         filters.
184         */
185        public BusinessQuery filterCategories(BusinessCategory... categories) {
186                this.categories = categories.clone();
187                this.hasCategories = true;
188                return this;
189        }
190
191        /**
192         * Filters the results by categories given by their alias (e.g. General
193         * Dentistry would be "generaldentistry") - so only results that businesses
194         * that are in the category will be returned. An arbitrary number of aliases
195         * can be given. A complete list of all the categories and their aliases is
196         * at <a
197         * href="http://www.yelp.com/developers/documentation/all_category_list"
198         * >http://www.yelp.com/developers/documentation/all_category_list</a>.
199         * 
200         * @param categories
201         * @return This BusinessQuery instance, allowing you to chain together other
202         *         filters.
203         */
204        public BusinessQuery filterCategories(String... categories) {
205                this.categories = new BusinessCategory[categories.length];
206                for (int i = 0; i < categories.length; i++) {
207                        this.categories[i] = BusinessCategory.getFromAlias(categories[i],
208                                        "");
209                }
210                this.hasCategories = true;
211                return this;
212        }
213
214        /**
215         * Remove any categories that are being filtered.
216         * 
217         * @return This BusinessQuery instance, allowing you to chain together other
218         *         filters.
219         */
220        public BusinessQuery removeCategoriesFilter() {
221                this.hasCategories = false;
222                return this;
223        }
224
225        /**
226         * Remove a list of categories from the filter.
227         * 
228         * @param categories
229         * @return This BusinessQuery instance, allowing you to chain together other
230         *         filters.
231         */
232        public BusinessQuery removeCategoriesFilter(BusinessCategory... categories) {
233                List<BusinessCategory> removingCategories = Arrays.asList(categories);
234                List<BusinessCategory> entireCategories = Arrays
235                                .asList(this.categories);
236                entireCategories.removeAll(removingCategories);
237                this.categories = entireCategories
238                                .toArray(new BusinessCategory[entireCategories.size()]);
239                if (entireCategories.isEmpty()) {
240                        this.hasCategories = false;
241                }
242                return this;
243        }
244
245        /**
246         * Remove a list of categories from the filter, given as alias strings.
247         * 
248         * @param categories
249         * @return This BusinessQuery instance, allowing you to chain together other
250         *         filters.
251         */
252        public BusinessQuery removeCategoriesFilter(String... categories) {
253                List<BusinessCategory> removingCategories = new ArrayList<BusinessCategory>();
254                for (int i = 0; i < categories.length; i++) {
255                        removingCategories.add(BusinessCategory.getFromAlias(categories[i],
256                                        ""));
257                }
258                List<BusinessCategory> entireCategories = Arrays
259                                .asList(this.categories);
260                entireCategories.removeAll(removingCategories);
261                this.categories = entireCategories
262                                .toArray(new BusinessCategory[entireCategories.size()]);
263                if (entireCategories.isEmpty()) {
264                        this.hasCategories = false;
265                }
266                return this;
267        }

396        /**
397         * Remove any "pagination". This is the same as removing a result limit and any offset.

402        public BusinessQuery removePagination() {

405                return this;

408        /**
409         * This method is intended for internal use only.<Br>
410         * <br>
411         * 
412         * Converts the internal list of filters into a dictionary.
413         * 
414         * @return
415         */
416        public HashMap<String, String> getFilters() {
417                HashMap<String, String> filters = new HashMap<String, String>();
418
419                filters.putAll(this.location.getFilter());
420
421                if (this.hasTerm) {
422                        filters.put("term", this.term);
423                }
424
425                if (this.hasLimit) {
426                        filters.put("limit", "" + this.limit);
427                }
428
429                if (this.hasOffset) {
430                        filters.put("offset", "" + this.offset);
431                }
432
433                if (this.hasCategories && this.categories.length >= 1) {
434                        StringBuilder csl = new StringBuilder();
435                        csl.append(this.categories[0]);
436                        if (this.categories.length > 1) {
437                                for (int i = 1; i < this.categories.length; i++) {
438                                        csl.append("," + this.categories[i]);
439                                }
440                        }
441                        filters.put("category_filter", csl.toString());
442                }
443
444                if (this.hasRadius) {
445                        filters.put("radius_filter", "" + this.radius);
446                }
447
448                if (this.hasDeals) {
449                        filters.put("deals_filter", (this.deals ? "1" : "0"));
450                }
451
452                if (this.hasSortMode) {
453                        filters.put("sort", "" + this.sortMode.ordinal());
454                }
455
456                return filters;
457        }
