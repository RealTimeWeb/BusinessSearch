001package realtimeweb.businesssearch.main;
002
003/**
004 * An abstract version of the Business Searchers. The implementations are:
005 * <ul>
006 * <li>JsonBusinessSearch: Returns json-formatted strings.</li>
007 * <li>StructuredBusinessSearch: Returns maps.</li>
008 * <li>BusinessSearch: Returns well-structured classes.</li>
009 * </ul> 
010 * @author acbart
011 *
012 */
013public interface AbstractBusinessSearch {
014        /**
015         * Connects an existing Business Searcher to the online Business Search service.
016         * @param consumerKey
017         * @param consumerSecret
018         * @param token
019         * @param tokenSecret
020         */
021        public void connect(String consumerKey, String consumerSecret, String token, String tokenSecret);
022
023        /**
024         * Establishes that data should be retrieved locally.
025         * 
026         * If data is being retrieved locally, you must be sure that your parameters
027         * match locally stored data. Otherwise, you will get nothing in return.
028         * @param spoofing
029         */
030        public void disconnect();
031
032}
