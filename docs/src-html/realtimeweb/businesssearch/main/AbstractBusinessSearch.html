001package realtimeweb.businesssearch.main;
002
003/**
004 * An abstract version of the Business Searchers. The implementations are:
005 * <ul>
006 * <li>{@link realtimeweb.businesssearch.json.JsonBusinessSearch
007 * JsonBusinessSearch}: Returns json-formatted strings.</li>
008 * <li>{@link realtimeweb.businesssearch.structured.StructuredBusinessSearch
009 * StructuredBusinessSearch}: Returns maps.</li>
010 * <li>{@link realtimeweb.businesssearch.regular.BusinessSearch BusinessSearch}:
011 * Returns well-structured classes.</li>
012 * </ul>
013 * 
014 * @author acbart
015 * 
016 */
017public interface AbstractBusinessSearch {
018        /**
019         * Connects an existing Business Searcher to the online Business Search
020         * service.
021         * 
022         * @param consumerKey
023         * @param consumerSecret
024         * @param token
025         * @param tokenSecret
026         */
027        public void connect(String consumerKey, String consumerSecret,
028                        String token, String tokenSecret);
029
030        /**
031         * Establishes that data should be retrieved locally.
032         * 
033         * If data is being retrieved locally, you must be sure that your parameters
034         * match locally stored data. Otherwise, you will get nothing in return.
035         * 
036         * @param spoofing
037         */
038        public void disconnect();
039
040}
