003/**
004 * If the mode is BEST_MATCHED, a search may retrieve an additional 20
005 * businesses past the initial limit of the first 20 results. This is done by
006 * specifying an offset and limit of 20. Sort by distance is only supported for
007 * a location or geographic search. The rating sort is not strictly sorted by
008 * the rating value, but by an adjusted rating value that takes into account the
009 * number of ratings, similar to a bayesian average. This is so a business with
010 * 1 rating of 5 stars doesn't immediately jump to the top.
011 * 
012 * @author acbart
013 * 
014 */
015public enum SortMode {
016        BEST_MATCHED, DISTANCE, HIGHEST_RATED;
017
018        /**
019         * Returns a human-readable version of this Sort Mode.
020         */
021        public String toString() {
022                switch (this) {
023                default:
024                case BEST_MATCHED:
025                        return "Best Matched";
026                case DISTANCE:
027                        return "Distance";
028                case HIGHEST_RATED:
029                        return "Highest Rated";
030                }
031        }
032}
