001package realtimeweb.businesssearch.regular;
002
003import realtimeweb.businesssearch.exceptions.BusinessSearchException;
004
005/**
006 * A listener for handling data received about a specific Business. On success,
007 * an instance of a Business will be passed to the businessDataCompleted method,
008 * which must be overridden in any implementing classes. On failure, an
009 * exception is passed to the businessDataFailed method, which must also be
010 * overridden in any implementing class.
011 * 
012 * @author acbart
013 * 
014 */
015public interface BusinessDataListener {
016        /**
017         * The method that should be overridden to handle the Business instance.
018         * 
019         * @param businessData
020         */
021        public abstract void businessDataCompleted(Business business);
022
023        /**
024         * The method that should be overridden to handle an exception that occurred
025         * while getting the Business Instance
026         * 
027         * @param exception
028         */
029        public abstract void businessDataFailed(BusinessSearchException exception);
030}
