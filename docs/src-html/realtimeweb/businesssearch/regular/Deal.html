001    package realtimeweb.businesssearch.regular;
002    
003    import java.util.ArrayList;
004    import java.util.HashMap;
005    import java.util.Iterator;
006    import java.util.List;
007    
008    /**
009     * A Deal is an offer made by the Business, such as a "2-for-1 Sale".
010     * 
011     * @author acbart
012     * 
013     */
014    public class Deal {
015            private String id;
016            private String title;
017            private String url;
018            private String imageUrl;
019            private String currencyCode;
020            private int timeStart;
021            private int timeEnd;
022            private boolean isPopular;
023            private String details;
024            private String importantRestrictions;
025            private String additionalRestrictions;
026    
027            private List<DealOption> options;
028    
029            @SuppressWarnings("unchecked")
030            Deal(HashMap<String, Object> raw) {
031                    this.id = (String) raw.get("id");
032                    this.title = (String) raw.get("title");
033                    this.url = (String) raw.get("url");
034                    this.imageUrl = (String) raw.get("image_url");
035                    this.currencyCode = (String) raw.get("currency_code");
036                    this.timeStart = Integer.parseInt((raw.get("time_start").toString()));
037                    this.timeEnd = -1;
038                    if (raw.containsKey("time_end")) {
039                            this.timeEnd = Integer.parseInt((raw.get("time_end").toString()));
040                    }
041                    this.isPopular = false;
042                    if (raw.containsKey("this.isPopular")) {
043                            this.isPopular = (Boolean) raw.get("is_popular");
044                    }
045                    this.details = (String) raw.get("what_you_get");
046                    this.importantRestrictions = (String) raw.get("important_restrictions");
047                    this.additionalRestrictions = (String) raw
048                                    .get("additional_restrictions");
049    
050                    this.options = new ArrayList<DealOption>();
051                    if (raw.containsKey("options")) {
052                            Iterator<HashMap<String, Object>> d = ((ArrayList<HashMap<String, Object>>) raw
053                                            .get("options")).iterator();
054                            while (d.hasNext()) {
055                                    this.options.add(new DealOption(d.next()));
056                            }
057                    }
058            }
059    
060            /**
061             * Returns a string with information about any additional restrictions.
062             * 
063             * @return the additionalRestrictions
064             */
065            public String getAdditionalRestrictions() {
066                    return additionalRestrictions;
067            }
068    
069            /**
070             * Returns the <a href='http://en.wikipedia.org/wiki/ISO_4217'>ISO-4217
071             * Currency Code</a>
072             * 
073             * @return the currencyCode
074             */
075            public String getCurrencyCode() {
076                    return currencyCode;
077            }
078    
079            /**
080             * Provides additional details for the Deal, separated by newlines
081             * 
082             * @return the details
083             */
084            public String getDetails() {
085                    return details;
086            }
087    
088            /**
089             * A unique identifier for the Deal.
090             * 
091             * @return the id
092             */
093            public String getId() {
094                    return id;
095            }
096    
097            /**
098             * A URL of the image for the deal.
099             * 
100             * @return the imageUrl
101             */
102            public String getImageUrl() {
103                    return imageUrl;
104            }
105    
106            /**
107             * Provides any important restrictions for the Deal, separated by newlines
108             * 
109             * @return the importantRestrictions
110             */
111            public String getImportantRestrictions() {
112                    return importantRestrictions;
113            }
114    
115            /**
116             * Provides the options for this Deal. If there are no options, an empty
117             * list is returned.
118             * 
119             * @return the options
120             */
121            public List<DealOption> getOptions() {
122                    return options;
123            }
124    
125            /**
126             * Returns when the deal ends as a Unix Timestamp. If the deal does not end,
127             * returns -1.
128             * 
129             * @return the timeEnd
130             */
131            public int getTimeEnd() {
132                    return timeEnd;
133            }
134    
135            /**
136             * Returns when the deal started as a Unix Timestamp.
137             * 
138             * @return the timeStart
139             */
140            public int getTimeStart() {
141                    return timeStart;
142            }
143    
144            /**
145             * Returns the title of the deal.
146             * 
147             * @return the title
148             */
149            public String getTitle() {
150                    return title;
151            }
152    
153            /**
154             * Returns the URL of the deal's page.
155             * 
156             * @return the url
157             */
158            public String getUrl() {
159                    return url;
160            }
161    
162            /**
163             * Returns whether the deal is popular. If popularity information was not
164             * available, false is returned.
165             * 
166             * @return the isPopular
167             */
168            public boolean isPopular() {
169                    return isPopular;
170            }
171    }
