001    package realtimeweb.businesssearch.regular;
002    
003    import realtimeweb.businesssearch.exceptions.BusinessSearchException;
004    
005    /**
006     * A listener for handling data received about a list of businesses. On success,
007     * an instance of a SearchResponse containing all the information about the List
008     * of Businesses will be passed to the businessSearchCompleted method, which
009     * must be overridden in any implementing classes. On failure, an exception is
010     * passed to the businessSearchFailed method, which must also be overridden in
011     * any implementing class.
012     * 
013     * @author acbart
014     * 
015     */
016    public interface BusinessSearchListener {
017            /**
018             * The method that should be overridden to handle the SearchResponse.
019             * 
020             * @param businessData
021             */
022            public abstract void businessSearchCompleted(SearchResponse searchResponse);
023    
024            /**
025             * The method that should be overridden to handle an exception that occurred
026             * while getting the SearchResponse.
027             * 
028             * @param exception
029             */
030            public abstract void businessSearchFailed(BusinessSearchException exception);
031    }
