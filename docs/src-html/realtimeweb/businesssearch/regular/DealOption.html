001package realtimeweb.businesssearch.regular;
002
003import java.util.HashMap;
004
005/**
006 * A DealOption is an item in a Deal.
007 * 
008 * @author acbart
009 * 
010 */
011public class DealOption {
012        private String title;
013        private String purchaseUrl;
014        private int price; // in cents
015        private String priceFormatted;
016        private int originalPrice;
017        private String originalPriceFormatted;
018        private boolean isQuantityLimited;
019        private int remainingCount;
020
021        DealOption(HashMap<String, Object> raw) {
022                this.title = (String) raw.get("title");
023                this.purchaseUrl = (String) raw.get("purchase_url");
024                this.price = Integer.parseInt((raw.get("price").toString()));
025                this.priceFormatted = (String) raw.get("formatted_price");
026                this.originalPrice = Integer.parseInt((raw.get("original_price")
027                                .toString()));
028                this.originalPriceFormatted = (String) raw
029                                .get("formatted_original_price");
030                this.isQuantityLimited = (boolean) raw.get("is_quantity_limited");
031                this.remainingCount = -1;
032                if (raw.containsKey("remaining_count")) {
033                        this.remainingCount = Integer.parseInt((raw.get("remaining_count")
034                                        .toString()));
035                }
036        }
037
038        /**
039         * Returns the original price of this option, before the deal is applied, in
040         * cents.
041         * 
042         * @return the originalPrice
043         */
044        public int getOriginalPrice() {
045                return originalPrice;
046        }
047
048        /**
049         * Returns the original price of this option, before the deal is applied, as
050         * a nicely formatted string.
051         * 
052         * @return the originalPriceFormatted
053         */
054        public String getOriginalPriceFormatted() {
055                return originalPriceFormatted;
056        }
057
058        /**
059         * Returns the price offered by this deal, in cents.
060         * 
061         * @return the price
062         */
063        public int getPrice() {
064                return price;
065        }
066
067        /**
068         * Returns the price offered by this deal as a nicely formatted string.
069         * 
070         * @return the priceFormatted
071         */
072        public String getPriceFormatted() {
073                return priceFormatted;
074        }
075
076        /**
077         * Returns the URL for this purchase option.
078         * 
079         * @return the purchaseUrl
080         */
081        public String getPurchaseUrl() {
082                return purchaseUrl;
083        }
084
085        /**
086         * Returns the remaining deal options available for purchase, if this deal
087         * is limited. If it is not limited, -1 is returned.
088         * 
089         * @return the remainingCount
090         */
091        public int getRemainingCount() {
092                return remainingCount;
093        }
094
095        /**
096         * Returns the title of this option.
097         * 
098         * @return the title
099         */
100        public String getTitle() {
101                return title;
102        }
103
104        /**
105         * Returns whether the quantity of this deal is limited. 
106         * 
107         * @return the isQuantityLimited
108         */
109        public boolean isQuantityLimited() {
110                return isQuantityLimited;
111        }
112}
