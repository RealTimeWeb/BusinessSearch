001package realtimeweb.businesssearch.regular;
002
003import java.util.Map;
004
005import realtimeweb.businesssearch.exceptions.BusinessSearchException;
006import realtimeweb.businesssearch.main.AbstractBusinessSearch;
007import realtimeweb.businesssearch.main.BusinessQuery;
008import realtimeweb.businesssearch.structured.StructuredBusinessDataListener;
009import realtimeweb.businesssearch.structured.StructuredBusinessSearch;
010import realtimeweb.businesssearch.structured.StructuredBusinessSearchListener;
011
012/**
013 * The StructuredBusinessSearch class is used to connect to the Yelp service and
014 * retrieve data as well-structured classes. All data is returned asynchronously
015 * - your program will continue to run while the data is being fetched, and the
016 * result will be sent to a callback function that you specify.<br>
017 * <br>
018 * 
019 * There will only be one instance of the StructuredBusinessSearch, because it
020 * follows the Singleton pattern. In order to use it, first get the instance of
021 * the class.
022 * 
023 * <pre>
024 * StructuredBusinessSearch sbs = StructuredBusinessSearch.getInstance();
025 * </pre>
026 * 
027 * Then you can either connect to the online service:
028 * 
029 * <pre>
030 * sbs.connect(&quot;key&quot;, &quot;secret&quot;, &quot;token&quot;, &quot;secrettoken&quot;);
031 * </pre>
032 * 
033 * or use the service offline:
034 * 
035 * <pre>
036 * sbs.setLocal();
037 * </pre>
038 * 
039 * @author acbart
040 * 
041 */
042public class BusinessSearch implements AbstractBusinessSearch {
043
044        private static BusinessSearch instance;
045
046        /**
047         * Retrieves the singleton instance of the JsonBusinessSearch.
048         * 
049         * @return
050         */
051        public static BusinessSearch getInstance() {
052                if (instance == null) {
053                        synchronized (BusinessSearch.class) {
054                                if (instance == null) {
055                                        instance = new BusinessSearch();
056                                }
057                        }
058                }
059                return instance;
060        }
061
062        /**
063         * Protected Constructor guards against instantiation.
064         */
065        protected BusinessSearch() {
066                structuredBusinessSearch = StructuredBusinessSearch.getInstance();
067        }
068
069        private StructuredBusinessSearch structuredBusinessSearch;
070
071        public void getBusinessData(final Business incompleteBusiness,
072                        final BusinessDataListener listener) {
073                this.structuredBusinessSearch.getBusinessData(incompleteBusiness.getId(),
074                                new StructuredBusinessDataListener() {
075
076                                        @Override
077                                        public void businessDataCompleted(
078                                                        Map<String, Object> business) {
079                                                incompleteBusiness.fillIn(business);
080                                                listener.businessDataCompleted(incompleteBusiness);
081                                        }
082
083                                        @Override
084                                        public void businessDataFailed(
085                                                        BusinessSearchException exception) {
086                                                listener.businessDataFailed(exception);
087                                        }
088
089                                });
090        }
091
092        /**
093         * Returns data about a business given its businessId (which is a sequence
094         * of characters, symbols, and numbers, e.g. "I_7cQmHxx6LokP9yr7Fx-w").
095         * 
096         * @param businessId
097         *            A string identifying the business.
098         * @param listener
099         *            a callback function that will consume the data (or error)
100         *            about the business.
101         */
102        public void getBusinessData(String businessId,
103                        final BusinessDataListener listener) {
104                this.structuredBusinessSearch.getBusinessData(businessId,
105                                new StructuredBusinessDataListener() {
106
107                                        @Override
108                                        public void businessDataCompleted(
109                                                        Map<String, Object> business) {
110                                                listener.businessDataCompleted(new Business(business));
111                                        }
112
113                                        @Override
114                                        public void businessDataFailed(
115                                                        BusinessSearchException exception) {
116                                                listener.businessDataFailed(exception);
117                                        }
118
119                                });
120        }
121
122        /**
123         * Queries the Yelp service for businesses that match the given
124         * BusinessQuery. Returns the data (or error) through the listener object.
125         * 
126         * @param query
127         *            A BusinessQuery object
128         * @param listener
129         *            A listener object that should contain methods for handling the
130         *            data and, optionally, any errors.
131         */
132        public void searchBusinesses(final BusinessQuery query,
133                        final BusinessSearchListener listener) {
134                this.structuredBusinessSearch.searchBusinesses(query,
135                                new StructuredBusinessSearchListener() {
136
137                                        @Override
138                                        public void businessSearchCompleted(
139                                                        Map<String, Object> response) {
140                                                listener.businessSearchCompleted(new SearchResponse(
141                                                                response));
142                                        }
143
144                                        @Override
145                                        public void businessSearchFailed(
146                                                        BusinessSearchException exception) {
147                                                listener.businessSearchFailed(exception);
148                                        }
149
150                                });
151        }
152
153        /**
154         * Establishes a connection to the online Business Search service.
155         * dataservice. This requires an internet connection.
156         * 
157         * Requires registration information from Yelp. To get your key go to
158         * http://www.yelp.com/developers
159         * 
160         * @param consumerKey
161         * @param consumerSecret
162         * @param token
163         * @param tokenSecret
164         */
165        public void connect(String consumerKey, String consumerSecret,
166                        String token, String tokenSecret) {
167                this.structuredBusinessSearch.connect(consumerKey, consumerSecret,
168                                token, tokenSecret);
169        }
170
171        /**
172         * Establishes that Business Search data should be retrieved locally. This
173         * does not require an internet connection.
174         * 
175         * If data is being retrieved locally, you must be sure that your parameters
176         * match locally stored data. Otherwise, you will get nothing in return.
177         * 
178         * @param local
179         */
180        public void disconnect() {
181                this.structuredBusinessSearch.disconnect();
182        }
183
184        public static void main(String[] args) {
185                final BusinessSearch bs = BusinessSearch.getInstance();
186                bs.disconnect();
187                /*
188                 * bs.searchBusinesses(new BusinessQuery("Blacksburg, VfA"), new
189                 * BusinessSearchListener() {
190                 * 
191                 * @Override public void onSuccess(SearchResponse searchResponse) {
192                 * System.out.println(searchResponse); }
193                 * 
194                 * });
195                 */
196                bs.getBusinessData("gillies-cuisine-blacksburg",
197                                new BusinessDataListener() {
198
199                                        @Override
200                                        public void businessDataCompleted(Business business) {
201                                                System.out.println(business);
202                                        }
203
204                                        @Override
205                                        public void businessDataFailed(
206                                                        BusinessSearchException exception) {
207                                                System.err.println(exception);
208                                        }
209
210                                });
211        }

213}
