001package realtimeweb.businesssearch.regular;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.HashMap;
006import java.util.Map;
007
008import realtimeweb.businesssearch.main.GeoAccuracy;
009import realtimeweb.businesssearch.main.Location;
010import realtimeweb.businesssearch.util.Util;
011
012/**
013 * A location that contains highly detailed information, including both the
014 * latitude/longitude coordinates and address.
015 * 
016 * @author acbart
017 * 
018 */
019public class DetailedLocation implements Location {
020
021        private double latitude;
022        private double longitude;
023        private ArrayList<String> address;
024        private ArrayList<String> displayAddress;
025        private String city;
026        private String stateCode;
027        private String postalCode;
028        private String countryCode;
029        private String crossStreets;
030        private ArrayList<String> neighborhoods;
031        private GeoAccuracy geoAccuracy;
032
033        /**
034         * Create a new DetailedLocation.
035         * 
036         * @param latitude
037         * @param longitude
038         * @param address
039         *            A list of fields describing the address, e.g.
040         *            ["1313 MockingBird Lane", "Blacksburg", "VA"]
041         * @param displayAddress
042         *            A list of detailed strings representing the postal address,
043         *            with more human-friendly information about the address.
044         * @param city
045         * @param stateCode
046         * @param postalCode
047         * @param countryCode
048         * @param crossStreets
049         * @param neighborhoods
050         * @param geoAccuracy
051         */
052        DetailedLocation(double latitude, double longitude,
053                        ArrayList<String> address, ArrayList<String> displayAddress,
054                        String city, String stateCode, String postalCode,
055                        String countryCode, String crossStreets,
056                        ArrayList<String> neighborhoods, GeoAccuracy geoAccuracy) {
057                super();
058                this.latitude = latitude;
059                this.longitude = longitude;
060                this.address = address;
061                this.displayAddress = displayAddress;
062                this.city = city;
063                this.stateCode = stateCode;
064                this.postalCode = postalCode;
065                this.countryCode = countryCode;
066                this.crossStreets = crossStreets;
067                this.neighborhoods = neighborhoods;
068                this.geoAccuracy = geoAccuracy;
069        }
070
071        @SuppressWarnings("unchecked")
072        DetailedLocation(HashMap<String, Object> hashMap) {
073                HashMap<String, Object> coordinates = (HashMap<String, Object>) hashMap
074                                .get("coordinate");
075                this.latitude = Double.parseDouble((coordinates.get("latitude")
076                                .toString()));
077                this.longitude = Double.parseDouble((coordinates.get("longitude")
078                                .toString()));
079
080                this.address = (ArrayList<String>) hashMap.get("address");
081                this.displayAddress = (ArrayList<String>) hashMap
082                                .get("display_address");
083
084                this.city = (String) hashMap.get("city");
085                this.stateCode = (String) hashMap.get("state_code");
086                this.postalCode = (String) hashMap.get("postal_code");
087                this.countryCode = (String) hashMap.get("country_code");
088                this.crossStreets = (String) hashMap.get("cross_streets");
089                this.neighborhoods = (ArrayList<String>) hashMap.get("neighborhoods");
090
091                this.geoAccuracy = GeoAccuracy.values()[(Integer.parseInt((hashMap
092                                .get("geo_accuracy").toString())))];
093        }
094
095        /**
096         * Returns a string representation of this Location based on it's address
097         * field.
098         */
099        public String toString() {
100                return Arrays.deepToString(this.address.toArray()).toString();
101        }
102
103        /**
104         * Returns the latitude of the location of this business.
105         * 
106         * @return the latitude
107         */
108        public double getLatitude() {
109                return latitude;
110        }
111
112        /**
113         * Returns the longitude of the geocoordinate of this location.
114         * 
115         * @return the longitude
116         */
117        public double getLongitude() {
118                return longitude;
119        }
120
121        /**
122         * Returns the address of this location as a list of strings.
123         * 
124         * @return the address
125         */
126        public ArrayList<String> getAddress() {
127                return address;
128        }
129
130        /**
131         * Returns a more human-readable address of this location as a list of
132         * strings.
133         * 
134         * @return the displayAddress
135         */
136        public ArrayList<String> getDisplayAddress() {
137                return displayAddress;
138        }
139
140        /**
141         * Returns the city that this location is in.
142         * 
143         * @return the city
144         */
145        public String getCity() {
146                return city;
147        }
148
149        /**
150         * Returns the <a href='http://en.wikipedia.org/wiki/ISO_3166-2'>two-letter
151         * State code</a> that this location is in (e.g. Virginia is "VA").
152         * 
153         * @return the stateCode
154         */
155        public String getStateCode() {
156                return stateCode;
157        }
158
159        /**
160         * Returns the postal (or zip) code of this location (e.g. Blacksburg is
161         * 24060). Remember that sometimes zip codes are represented as 9-characters
162         * instead of 5.
163         * 
164         * @return the postalCode
165         */
166        public String getPostalCode() {
167                return postalCode;
168        }
169
170        /**
171         * Returns the <a
172         * href='http://en.wikipedia.org/wiki/ISO_3166-1_alpha-2'>two-letter country
173         * code</a> for this location.
174         * 
175         * @return the countryCode
176         */
177        public String getCountryCode() {
178                return countryCode;
179        }
180
181        /**
182         * Returns the Cross streets for the business that this location describes.
183         * A cross street is the street closest to a location that intersects the
184         * locations's street.
185         * 
186         * @return the crossStreets
187         */
188        public String getCrossStreets() {
189                return crossStreets;
190        }
191
192        /**
193         * List that provides neighborhood(s) information for this location.
194         * 
195         * @return the neighborhoods
196         */
197        public ArrayList<String> getNeighborhoods() {
198                return neighborhoods;
199        }
200
201        /**
202         * Returns the {@link realtimeweb.businesssearch.main.GeoAccuracy GeoAccuracy} of this location.
203         * 
204         * @return the geoAccuracy
205         */
206        public GeoAccuracy getGeoAccuracy() {
207                return geoAccuracy;
208        }
209
210        /**
211         * <b>This method is for internal purposes only.</b><br><br>
212         * 
213         */
214        @Override
215        public Map<String, String> getFilter() {
216                HashMap<String, String> filter = new HashMap<String, String>();
217                String namedLocation = Util.join(this.address);
218                filter.put("location", namedLocation);
219                filter.put(
220                                "cll",
221                                Util.formatDecimals(this.latitude, 6) + ","
222                                                + Util.formatDecimals(this.longitude, 6));
223                return filter;
224        }
225}
