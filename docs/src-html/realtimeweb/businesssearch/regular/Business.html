012/**
013 * A Business object represents a Business in the real world. It has a large
014 * number of properties associated with it to describe the Business.
015 * 
016 * @author acbart
017 * 
018 */
019public class Business {
020
021        private String id;
022        private boolean isClaimed;
023        private boolean isClosed;
024        private String name;
025        private String imageUrl;
026        private String url;
027        private String mobileUrl;
028        private String phone;
029        private String displayPhone;
030        private int reviewCount;
031        private List<BusinessCategory> categories;
032        private double distance;
033        private float rating;
034        private String ratingImageUrl;
035        private String ratingImageUrlSmall;
036        private String ratingImageUrlLarge;
037        private String snippetText;
038        private String snippetImageUrl;
039        private DetailedLocation location;
040        private List<Deal> deals;
041        private List<GiftCertificate> giftCertificates;
042        private List<Review> reviews;

045        @SuppressWarnings("unchecked")
046        Business(Map<String, Object> raw) {
047                this.isComplete = true;
048                this.id = (String) raw.get("id");
049                this.isClaimed = (boolean) raw.get("is_claimed");
050                this.isClosed = (boolean) raw.get("is_closed");
051                this.name = (String) raw.get("name");
052                this.imageUrl = (String) raw.get("image_url");
053                this.url = (String) raw.get("url");
054                this.mobileUrl = (String) raw.get("mobile_url");
055                this.phone = (String) raw.get("phone");
056                this.displayPhone = (String) raw.get("display_phone");
057                this.reviewCount = Integer
058                                .parseInt((raw.get("review_count").toString()));
059
060                this.categories = new ArrayList<BusinessCategory>();
061                if (raw.get("categories") != null) {
062                        Iterator<ArrayList<String>> i = ((ArrayList<ArrayList<String>>) raw
063                                        .get("categories")).iterator();
064                        while (i.hasNext()) {
065                                this.categories.add(BusinessCategory.getFromAlias(i.next()));
066                        }
067                }
068
069                if (raw.containsKey("distance")) {
070                        this.distance = Double
071                                        .parseDouble((raw.get("distance").toString()));
072                } else {
073                        this.distance = -1;
074                }
075                this.rating = Float.parseFloat((raw.get("rating").toString()));
076
077                this.ratingImageUrl = (String) raw.get("rating_image_url");
078                this.ratingImageUrlSmall = (String) raw.get("rating_image_url_small");
079                this.ratingImageUrlLarge = (String) raw.get("rating_image_url_large");
080                this.snippetText = (String) raw.get("snippet_text");
081                this.snippetImageUrl = (String) raw.get("snippet_image_url");
082                this.location = new DetailedLocation(
083                                (HashMap<String, Object>) raw.get("location"));
084
085                this.deals = new ArrayList<Deal>();
086                if (raw.containsKey("deals")) {
087                        Iterator<HashMap<String, Object>> d = ((ArrayList<HashMap<String, Object>>) raw
088                                        .get("deals")).iterator();
089                        while (d.hasNext()) {
090                                this.deals.add(new Deal(d.next()));
091                        }
092                }
093
094                this.giftCertificates = new ArrayList<GiftCertificate>();
095                if (raw.containsKey("gift_certificates")) {
096                        Iterator<HashMap<String, Object>> g = ((ArrayList<HashMap<String, Object>>) raw
097                                        .get("gift_certificates")).iterator();
098                        while (g.hasNext()) {
099                                this.giftCertificates.add(new GiftCertificate(g.next()));
100                        }
101                }
102
103                this.reviews = new ArrayList<Review>();
104                if (raw.containsKey("reviews")) {
105                        Iterator<HashMap<String, Object>> r = ((ArrayList<HashMap<String, Object>>) raw
106                                        .get("reviews")).iterator();
107                        while (r.hasNext()) {
108                                this.reviews.add(new Review(r.next()));
109                        }
110                }
111        }

113        /**
114         * Creates a new String representation of this Business, which will include
115         * the businesses ID and Name.
116         * 
117         * @return A string representing the Business
118         */
119        public String toString() {
120                return (new PropertyString("Business")).add("id", this.id.toString())
121                                .add("name", this.name.toString()).toString();
122        }
