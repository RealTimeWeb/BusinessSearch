001package realtimeweb.businesssearch.regular;
002
003import java.util.ArrayList;
004import java.util.HashMap;
005import java.util.Iterator;
006import java.util.List;
007
008/**
009 * A GiftCertificate is something you can buy to give as a present to someone
010 * else, and will have the value of however much you bought.
011 * 
012 * @author acbart
013 * 
014 */
015public class GiftCertificate {
016        private String id;
017        private String url;
018        private String imageUrl;
019        private String currencyCode;
020        private String unusedBalances;
021        private List<GiftCertificateOption> options;
022
023        @SuppressWarnings("unchecked")
024        GiftCertificate(HashMap<String, Object> raw) {
025                this.id = (String) raw.get("id");
026                this.url = (String) raw.get("url");
027                this.imageUrl = (String) raw.get("image_url");
028                this.currencyCode = (String) raw.get("currency_code");
029                this.unusedBalances = (String) raw.get("unused_balance");
030
031                this.options = new ArrayList<GiftCertificateOption>();
032                if (raw.containsKey("options")) {
033                        Iterator<HashMap<String, Object>> o = ((ArrayList<HashMap<String, Object>>) raw
034                                        .get("options")).iterator();
035                        while (o.hasNext()) {
036                                this.options.add(new GiftCertificateOption(o.next()));
037                        }
038                }
039        }
040
041        /**
042         * Returns a unique identifier for this GiftCertificate.
043         * 
044         * @return the id
045         */
046        public String getId() {
047                return id;
048        }
049
050        /**
051         * Returns the Gift certificate landing page url.
052         * 
053         * @return the url
054         */
055        public String getUrl() {
056                return url;
057        }
058
059        /**
060         * Returns the Gift certificate image url.
061         * 
062         * @return the imageUrl
063         */
064        public String getImageUrl() {
065                return imageUrl;
066        }
067
068        /**
069         * Returns the <a href='http://en.wikipedia.org/wiki/ISO_4217'>ISO-4217
070         * Currency Code</a>
071         * 
072         * @return the currencyCode
073         */
074        public String getCurrencyCode() {
075                return currencyCode;
076        }
077
078        /**
079         * Returns how unused balances are returned - either as cash ("CASH") or
080         * store credit ("CREDIT").
081         * 
082         * @return the unusedBalances
083         */
084        public String getUnusedBalances() {
085                return unusedBalances;
086        }
087
088        /**
089         * Returns a list of Gift Certificate options that are available, or how
090         * much you can purchase a gift certificate for (e.g. $10, $20, $50, ...).
091         * 
092         * @return the options
093         */
094        public List<GiftCertificateOption> getOptions() {
095                return options;
096        }
097
098}
