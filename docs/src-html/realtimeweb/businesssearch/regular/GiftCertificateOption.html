001    package realtimeweb.businesssearch.regular;
002    
003    import java.util.HashMap;
004    
005    /**
006     * An option for a Gift Certificate - that is, a possible value for a gift
007     * certificate to have (for instance, Kohl's might have $10, $20, and $50 gift
008     * cards available).
009     * 
010     * @author acbart
011     * 
012     */
013    public class GiftCertificateOption {
014            private int price;
015    
016            /**
017             * This is an internal method. Please ignore it.
018             * 
019             * @param raw
020             */
021            GiftCertificateOption(HashMap<String, Object> raw) {
022                    this.price = Integer.parseInt((raw.get("price").toString()));
023                    this.formattedPrice = (String) raw.get("formatted_price");
024            }
025    
026            private String formattedPrice;
027    
028            /**
029             * Returns the price of this Gift Certificate Option in cents ($50 would be
030             * 5000). Cents are used instead of dollars because decimal values cannot be
031             * stored correctly in computers.
032             * 
033             * @return the price
034             */
035            public int getPrice() {
036                    return price;
037            }
038    
039            /**
040             * Returns a nicely-formatted string representation of this price (e.g.
041             * "$50").
042             * 
043             * @return the formattedPrice
044             */
045            public String getFormattedPrice() {
046                    return formattedPrice;
047            }
048    }
