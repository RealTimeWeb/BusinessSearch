001    package realtimeweb.businesssearch.regular;
002    
003    import java.util.ArrayList;
004    import java.util.HashMap;
005    import java.util.Iterator;
006    import java.util.List;
007    import java.util.Map;
008    
009    import realtimeweb.businesssearch.main.RectangularLocation;
010    import realtimeweb.businesssearch.util.PropertyString;
011    import realtimeweb.businesssearch.util.Util;
012    
013    /**
014     * A SearchResponse is returned by a search of businesses around a Location. It
015     * has a list of businesses, a suggested region of a map to display those
016     * businesses, and the total number of businesses found by the search (which
017     * will often be more than the list of businesses in the SearchResponse, since
018     * usually only the first 20 results are returned). In order to access these
019     * extra businesses, use the
020     * {@link realtimeweb.businesssearch.main.BusinessQuery#offsetResults(int)
021     * offsetResults} or
022     * {@link realtimeweb.businesssearch.main.BusinessQuery#paginate(int, int)
023     * paginate} filters in the
024     * {@link realtimeweb.businesssearch.main.BusinessQuery BusinessQuery}.
025     * 
026     * @author acbart
027     * 
028     */
029    public class SearchResponse {
030            private List<Business> businesses;
031            private RectangularLocation suggestedDisplayRegion;
032            private int totalFound;
033    
034            @SuppressWarnings("unchecked")
035            SearchResponse(Map<String, Object> response) {
036                    this.totalFound = Integer.parseInt((response.get("total").toString()));
037                    HashMap<String, Object> region = (HashMap<String, Object>) response
038                                    .get("region");
039                    this.suggestedDisplayRegion = RectangularLocation.fromCenter(region);
040    
041                    this.businesses = new ArrayList<Business>();
042                    Iterator<HashMap<String, Object>> i = ((ArrayList<HashMap<String, Object>>) response
043                                    .get("businesses")).iterator();
044                    while (i.hasNext()) {
045                            HashMap<String, Object> value = i.next();
046                            Business b = new Business(value);
047                            b.setIsComplete(false);
048                            this.businesses.add(b);
049                    }
050            }
051    
052            /**
053             * Returns a string representation of the businesses found and the total
054             * number of businesses found.
055             */
056            public String toString() {
057                    ArrayList<String> businessStrings = new ArrayList<String>();
058                    for (Business b : businesses) {
059                            businessStrings.add(b.toString());
060                    }
061                    return (new PropertyString("SearchResponse"))
062                                    .add("totalFound", "" + totalFound)
063                                    .add("businesses", "[" + Util.join(businessStrings) + "]")
064                                    .toString();
065            }
066    
067            /**
068             * Returns a list of the businesses found.
069             * 
070             * @return the businesses
071             */
072            public List<Business> getBusinesses() {
073                    return businesses;
074            }
075    
076            /**
077             * Returns a RectangularLocation describing the suggested region to display
078             * the results of the search.
079             * 
080             * @return the suggestedDisplayRegion
081             */
082            public RectangularLocation getSuggestedDisplayRegion() {
083                    return suggestedDisplayRegion;
084            }
085    
086            /**
087             * Returns the total number of businesses found, which may be greater than
088             * the businesses returned by {@link #getBusinesses}. In order to access
089             * these extra businesses, use the
090             * {@link realtimeweb.businesssearch.main.BusinessQuery#offsetResults(int) offsetResults} or
091             * {@link realtimeweb.businesssearch.main.BusinessQuery#paginate(int, int) paginate} filters in the
092             * {@link realtimeweb.businesssearch.main.BusinessQuery BusinessQuery}.
093             * 
094             * @return the totalFound
095             */
096            public int getTotalFound() {
097                    return totalFound;
098            }
099    }
