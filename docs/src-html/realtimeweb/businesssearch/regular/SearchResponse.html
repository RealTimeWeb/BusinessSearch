001package realtimeweb.businesssearch.regular;
002
003import java.util.ArrayList;
004import java.util.HashMap;
005import java.util.Iterator;
006import java.util.List;
007import java.util.Map;
008
009import realtimeweb.businesssearch.main.RectangularLocation;
010import realtimeweb.businesssearch.util.PropertyString;
011import realtimeweb.businesssearch.util.Util;
012
013/**
014 * A SearchResponse is returned by a search of businesses around a Location. It
015 * has a list of businesses, a suggested region of a map to display those
016 * businesses, and the total number of businesses found by the search (which
017 * will often be more than the list of businesses in the SearchResponse, since
018 * usually only the first 20 results are returned). In order to access these
019 * extra businesses, use the {@link main.BusinessQuery#offsetResults(int)
020 * offsetResults} or {@link main.BusinessQuery#paginate(int, int) paginate}
021 * filters in the {@link main.BusinessQuery BusinessQuery}.
022 * 
023 * @author acbart
024 * 
025 */
026public class SearchResponse {
027        private List<Business> businesses;
028        private RectangularLocation suggestedDisplayRegion;
029        private int totalFound;
030
031        @SuppressWarnings("unchecked")
032        SearchResponse(Map<String, Object> response) {
033                this.totalFound = Integer.parseInt((response.get("total").toString()));
034                HashMap<String, Object> region = (HashMap<String, Object>) response
035                                .get("region");
036                this.suggestedDisplayRegion = RectangularLocation.fromCenter(region);
037
038                this.businesses = new ArrayList<Business>();
039                Iterator<HashMap<String, Object>> i = ((ArrayList<HashMap<String, Object>>) response
040                                .get("businesses")).iterator();
041                while (i.hasNext()) {
042                        HashMap<String, Object> value = i.next();
043                        Business b = new Business(value);
044                        b.setIsComplete(false);
045                        this.businesses.add(b);
046                }
047        }
048
049        /**
050         * Returns a string representation of the businesses found and the total
051         * number of businesses found.
052         */
053        public String toString() {
054                ArrayList<String> businessStrings = new ArrayList<String>();
055                for (Business b : businesses) {
056                        businessStrings.add(b.toString());
057                }
058                return (new PropertyString("SearchResponse"))
059                                .add("totalFound", "" + totalFound)
060                                .add("businesses", "[" + Util.join(businessStrings) + "]")
061                                .toString();
062        }
063
064        /**
065         * Returns a list of the businesses found.
066         * 
067         * @return the businesses
068         */
069        public List<Business> getBusinesses() {
070                return businesses;
071        }
072
073        /**
074         * Returns a RectangularLocation describing the suggested region to display
075         * the results of the search.
076         * 
077         * @return the suggestedDisplayRegion
078         */
079        public RectangularLocation getSuggestedDisplayRegion() {
080                return suggestedDisplayRegion;
081        }
082
083        /**
084         * Returns the total number of businesses found, which may be greater than
085         * the businesses returned by {@link #getBusinesses}. In order to access
086         * these extra businesses, use the
087         * {@link main.BusinessQuery#offsetResults(int) offsetResults} or
088         * {@link main.BusinessQuery#paginate(int, int) paginate} filters in the
089         * {@link main.BusinessQuery BusinessQuery}.
090         * 
091         * @return the totalFound
092         */
093        public int getTotalFound() {
094                return totalFound;
095        }
096}
