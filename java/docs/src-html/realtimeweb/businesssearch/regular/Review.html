001    package realtimeweb.businesssearch.regular;
002    
003    import java.util.HashMap;
004    
005    /**
006     * A Review contains user-supplied information about a Business.
007     * @author acbart
008     *
009     */
010    public class Review {
011            private String id;
012            private int rating;
013            private String ratingImageUrl;
014            private String ratingImageUrlSmall;
015            private String ratingImageUrlLarge;
016            private String excerpt;
017            private int timeCreated;
018            private User user;
019    
020            
021            @SuppressWarnings("unchecked")
022            Review(HashMap<String, Object> raw) {
023                    this.id = (String) raw.get("id");
024                    this.rating = Integer.parseInt((raw.get("rating").toString()));
025                    this.ratingImageUrl = (String) raw.get("rating_image_url");
026                    this.ratingImageUrlSmall = (String) raw.get("rating_image_small_url");
027                    this.ratingImageUrlLarge = (String) raw.get("rating_image_large_url");
028                    this.excerpt = (String) raw.get("excerpt");
029                    this.timeCreated = Integer.parseInt((raw.get("time_created").toString()));
030                    this.user = new User((HashMap<String, Object>)raw.get("user"));
031            }
032    
033            /**
034             * Returns a unique identifier for the Review.
035             * @return the id
036             */
037            public String getId() {
038                    return id;
039            }
040            /**
041             * Returns a value from 1-5 rating the Business.
042             * @return the rating
043             */
044            public int getRating() {
045                    return rating;
046            }
047            /**
048             * Returns a URL that links to an image of stars for this business (size = 84x17).
049             * @return the ratingImageUrl
050             */
051            public String getRatingImageUrl() {
052                    return ratingImageUrl;
053            }
054            /**
055             * Returns a URL that links to a smaller image of stars for this business (size = 50x10).
056             * @return the ratingImageUrlSmall
057             */
058            public String getRatingImageUrlSmall() {
059                    return ratingImageUrlSmall;
060            }
061            /**
062             * Returns a URL that links to a larger image of stars for this business (size = 166x30).
063             * @return the ratingImageUrlLarge
064             */
065            public String getRatingImageUrlLarge() {
066                    return ratingImageUrlLarge;
067            }
068            /**
069             * Returns an excerpt of the text of this review. 
070             * @return the excerpt
071             */
072            public String getExcerpt() {
073                    return excerpt;
074            }
075            /**
076             * Returns a Unix Timestamp indicating the date/time that this Review was created.
077             * @return the timeCreated
078             */
079            public int getTimeCreated() {
080                    return timeCreated;
081            }
082            /**
083             * Returns the User who wrote this review.
084             * @return the user
085             */
086            public User getUser() {
087                    return user;
088            }       
089    }
