001    package realtimeweb.businesssearch.regular;
002    
003    import java.util.Map;
004    
005    import realtimeweb.businesssearch.exceptions.BusinessSearchException;
006    import realtimeweb.businesssearch.main.AbstractBusinessSearch;
007    import realtimeweb.businesssearch.main.BusinessQuery;
008    import realtimeweb.businesssearch.structured.StructuredBusinessDataListener;
009    import realtimeweb.businesssearch.structured.StructuredBusinessSearch;
010    import realtimeweb.businesssearch.structured.StructuredBusinessSearchListener;
011    
012    /**
013     * The StructuredBusinessSearch class is used to connect to the Yelp service and
014     * retrieve data as well-structured classes. All data is returned asynchronously
015     * - your program will continue to run while the data is being fetched, and the
016     * result will be sent to a callback function that you specify.<br>
017     * <br>
018     * 
019     * There will only be one instance of the StructuredBusinessSearch, because it
020     * follows the Singleton pattern. In order to use it, first get the instance of
021     * the class.
022     * 
023     * <pre>
024     * StructuredBusinessSearch sbs = StructuredBusinessSearch.getInstance();
025     * </pre>
026     * 
027     * You can immediately use the service offline, or you can connect to the online
028     * service:
029     * 
030     * <pre>
031     * sbs.connect(&quot;key&quot;, &quot;secret&quot;, &quot;token&quot;, &quot;secrettoken&quot;);
032     * </pre>
033     * 
034     * To stop using your connection:
035     * 
036     * <pre>
037     * sbs.disconnect();
038     * </pre>
039     * 
040     * @author acbart
041     * 
042     */
043    public class BusinessSearch implements AbstractBusinessSearch {
044    
045            private static BusinessSearch instance;
046    
047            /**
048             * Retrieves the singleton instance of the JsonBusinessSearch.
049             * 
050             * @return
051             */
052            public static BusinessSearch getInstance() {
053                    if (instance == null) {
054                            synchronized (BusinessSearch.class) {
055                                    if (instance == null) {
056                                            instance = new BusinessSearch();
057                                    }
058                            }
059                    }
060                    return instance;
061            }
062    
063            /**
064             * Protected Constructor guards against instantiation.
065             */
066            protected BusinessSearch() {
067                    structuredBusinessSearch = StructuredBusinessSearch.getInstance();
068            }
069    
070            private StructuredBusinessSearch structuredBusinessSearch;
071    
072            /**
073             * Given a Business, fills in any data that might be missing from it. This
074             * is useful for the Businesses returned in a {@link SearchResponse} by a
075             * {@link #searchBusinesses}.
076             * 
077             * @param incompleteBusiness
078             * @param listener
079             */
080            public void getBusinessData(final Business incompleteBusiness,
081                            final BusinessDataListener listener) {
082                    this.structuredBusinessSearch.getBusinessData(
083                                    incompleteBusiness.getId(),
084                                    new StructuredBusinessDataListener() {
085    
086                                            @Override
087                                            public void businessDataCompleted(
088                                                            Map<String, Object> business) {
089                                                    incompleteBusiness.fillIn(business);
090                                                    listener.businessDataCompleted(incompleteBusiness);
091                                            }
092    
093                                            @Override
094                                            public void businessDataFailed(
095                                                            BusinessSearchException exception) {
096                                                    listener.businessDataFailed(exception);
097                                            }
098    
099                                    });
100            }
101    
102            /**
103             * Returns data about a business given its businessId (which is a sequence
104             * of characters, symbols, and numbers, e.g. "I_7cQmHxx6LokP9yr7Fx-w").
105             * 
106             * @param businessId
107             *            A string identifying the business.
108             * @param listener
109             *            a callback function that will consume the data (or error)
110             *            about the business.
111             */
112            public void getBusinessData(String businessId,
113                            final BusinessDataListener listener) {
114                    this.structuredBusinessSearch.getBusinessData(businessId,
115                                    new StructuredBusinessDataListener() {
116    
117                                            @Override
118                                            public void businessDataCompleted(
119                                                            Map<String, Object> business) {
120                                                    listener.businessDataCompleted(new Business(business));
121                                            }
122    
123                                            @Override
124                                            public void businessDataFailed(
125                                                            BusinessSearchException exception) {
126                                                    listener.businessDataFailed(exception);
127                                            }
128    
129                                    });
130            }
131    
132            /**
133             * Queries the Yelp service for businesses that match the given
134             * BusinessQuery. Returns the data (or error) through the listener object.
135             * 
136             * @param query
137             *            A BusinessQuery object
138             * @param listener
139             *            A listener object that should contain methods for handling the
140             *            data and, optionally, any errors.
141             */
142            public void searchBusinesses(final BusinessQuery query,
143                            final BusinessSearchListener listener) {
144                    this.structuredBusinessSearch.searchBusinesses(query,
145                                    new StructuredBusinessSearchListener() {
146    
147                                            @Override
148                                            public void businessSearchCompleted(
149                                                            Map<String, Object> response) {
150                                                    listener.businessSearchCompleted(new SearchResponse(
151                                                                    response));
152                                            }
153    
154                                            @Override
155                                            public void businessSearchFailed(
156                                                            BusinessSearchException exception) {
157                                                    listener.businessSearchFailed(exception);
158                                            }
159    
160                                    });
161            }
162    
163            /**
164             * Establishes a connection to the online Business Search service.
165             * dataservice. This requires an internet connection.
166             * 
167             * Requires registration information from Yelp. To get your key go to
168             * <a href='http://www.yelp.com/developers'>http://www.yelp.com/developers</a>
169             * 
170             * @param consumerKey
171             * @param consumerSecret
172             * @param token
173             * @param tokenSecret
174             */
175            public void connect(String consumerKey, String consumerSecret,
176                            String token, String tokenSecret) {
177                    this.structuredBusinessSearch.connect(consumerKey, consumerSecret,
178                                    token, tokenSecret);
179            }
180    
181            /**
182             * Establishes that Business Search data should be retrieved locally. This
183             * does not require an internet connection.<br><br>
184             * 
185             * If data is being retrieved locally, you must be sure that your parameters
186             * match locally stored data. Otherwise, you will get nothing in return.
187             * 
188             * @param local
189             */
190            public void disconnect() {
191                    this.structuredBusinessSearch.disconnect();
192            }
193    
194            public static void main(String[] args) {
195                    final BusinessSearch bs = BusinessSearch.getInstance();
196                    bs.disconnect();
197                    /*
198                     * bs.searchBusinesses(new BusinessQuery("Blacksburg, VfA"), new
199                     * BusinessSearchListener() {
200                     * 
201                     * @Override public void onSuccess(SearchResponse searchResponse) {
202                     * System.out.println(searchResponse); }
203                     * 
204                     * });
205                     */
206                    bs.getBusinessData("gillies-cuisine-blacksburg",
207                                    new BusinessDataListener() {
208    
209                                            @Override
210                                            public void businessDataCompleted(Business business) {
211                                                    System.out.println(business);
212                                            }
213    
214                                            @Override
215                                            public void businessDataFailed(
216                                                            BusinessSearchException exception) {
217                                                    System.err.println(exception);
218                                            }
219    
220                                    });
221            }
222    
223    }
