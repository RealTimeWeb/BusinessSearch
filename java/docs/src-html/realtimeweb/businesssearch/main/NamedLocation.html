001    package realtimeweb.businesssearch.main;
002    
003    import java.util.HashMap;
004    import java.util.Map;
005    
006    import realtimeweb.businesssearch.util.Util;
007    
008    /**
009     * A location specified primarily by an address - usually describing a larger
010     * area (such as "Blacksburg, VA"). Optionally, a latitude and longitude can be
011     * specified to disambiguate the region (for example, if the address is
012     * "Newark", then the latitude and longitude could be given to make it clear
013     * that we're talking about the Newark in Delaware and not the one in New
014     * Jersey.
015     * 
016     * @author acbart
017     * 
018     */
019    public class NamedLocation implements Location {
020            private String address;
021            private double latitudeHint;
022            private double longitudeHint;
023            private boolean hinting;
024    
025            /**
026             * Create a new NamedLocation based on an address. Usually the area
027             * described by the address is broad - for example, "Blacksburg, VA" would
028             * be a better choice than "1313 Mockingbird Lane, Blacksburg VA".
029             * 
030             * @param address
031             */
032            public NamedLocation(String address) {
033                    this.setAddress(address);
034                    this.hinting = false;
035            }
036    
037            /**
038             * Create a new NamedLocation based on an address (which usually describes a
039             * broad area), and also specifying a latitude and longitude to disambiguate
040             * the location. So, for instance, you can use:
041             * <code>NamedLocation("Newark", 39, 75)</code> to specify the Newark in
042             * Delaware, as opposed to the Newark in New Jersey.
043             * 
044             * @param address
045             * @param latitudeHint
046             * @param longitudeHint
047             */
048            public NamedLocation(String address, double latitudeHint,
049                            double longitudeHint) {
050                    this.setAddress(address);
051                    this.setHint(latitudeHint, longitudeHint);
052            }
053    
054            /**
055             * Returns the address specified for this location.
056             * 
057             * @return
058             */
059            public String getAddress() {
060                    return address;
061            }
062    
063            /**
064             * Sets the address for this location. The address usually describes a broad
065             * area, e.g. "Blacksburg VA".
066             * 
067             * @param address
068             */
069            public void setAddress(String address) {
070                    this.address = address;
071            }
072    
073            /**
074             * Removes the latitude/longitude hint for this location. Only the address
075             * will be used to specify the area.
076             */
077            public void removeHint() {
078                    this.hinting = false;
079            }
080    
081            /**
082             * Sets a latitude/longitude hint for the address, helping to disambiguate
083             * between similarly named areas. So, for instance, if the address was
084             * "Newark", specifying a latitude and longitude of (39,75) will specify the
085             * Newark in Delaware, as opposed to the Newark in New Jersey.
086             * 
087             * @param latitudeHint
088             * @param longitudeHint
089             */
090            public void setHint(double latitudeHint, double longitudeHint) {
091                    this.latitudeHint = latitudeHint;
092                    this.longitudeHint = longitudeHint;
093                    this.hinting = true;
094            }
095    
096            /**
097             * Returns the latitude that is being used to disambiguate the location.
098             * 
099             * @return
100             */
101            public double getLatitudeHint() {
102                    return latitudeHint;
103            }
104    
105            /**
106             * Returns the longitude that is being used to disambiguate the location.
107             * 
108             * @return
109             */
110            public double getLongitudeHint() {
111                    return longitudeHint;
112            }
113    
114            /**
115             * Returns whether a latitude and longitude is being used to disambiguate
116             * this address.
117             * 
118             * @return
119             */
120            public boolean hasHint() {
121                    return hinting;
122            }
123    
124            /**
125             * <b>This method is for internal use only. Please ignore it.</b><br>
126             * 
127             * Used to create a dictionary of key-value pairs for this location.
128             */
129            @Override
130            public Map<String, String> getFilter() {
131                    HashMap<String, String> filter = new HashMap<String, String>();
132                    filter.put("location", this.address);
133                    if (this.hinting) {
134                            filter.put("cll", Util.formatDecimals(this.latitudeHint, 6) + ","
135                                            + Util.formatDecimals(this.longitudeHint, 6));
136                    }
137                    return filter;
138            }
139    }
