001    package realtimeweb.businesssearch.main;
002    
003    import java.util.HashMap;
004    import java.util.Map;
005    
006    import realtimeweb.businesssearch.util.Util;
007    
008    /**
009     * A RectangularLocation is a location specified by two latitude/longitude
010     * pairs. The first pair is in the southwest (bottom-left) and the second is in
011     * the northeast (top-right).
012     * 
013     * @author acbart
014     * 
015     */
016    public class RectangularLocation implements Location {
017            private double south;
018            private double west;
019            private double north;
020            private double east;
021    
022            /**
023             * Specifies a location via a rectangular bounding box using two
024             * latitude/longitude pairs.
025             * 
026             * @param south
027             *            The latitude-component of the lower-leftmost part of the box
028             * @param west
029             *            The longitude-component of the lower-leftmost part of the box
030             * @param north
031             *            The latitude-component of the upper-rightmost part of the box
032             * @param east
033             *            The longitude-component of the upper-rightmost part of the box
034             */
035            public RectangularLocation(double south, double west, double north,
036                            double east) {
037                    this.setSouth(south);
038                    this.setNorth(north);
039                    this.setWest(west);
040                    this.setEast(east);
041            }
042    
043            /**
044             * <b>
045             * This is an internal method. Please ignore it.</b><br><br>
046             * 
047             * A method for converting the Location to a representation that can be used
048             * to construct a query.
049             */
050            @Override
051            public Map<String, String> getFilter() {
052                    HashMap<String, String> filter = new HashMap<String, String>();
053                    String sw = Util.formatDecimals(this.south, 6) + ","
054                                    + Util.formatDecimals(this.west, 6);
055                    String ne = Util.formatDecimals(this.north, 6) + ","
056                                    + Util.formatDecimals(this.east, 6);
057                    filter.put("bounds", sw + "|" + ne);
058                    return filter;
059            }
060    
061            /**
062             * Modifies this RectangularLocation so that it is centered at the given
063             * latitude and longitude, with a rectangular area given by dimensions
064             * latitudeDelta by longitudeDelta.
065             * 
066             * @param latitude
067             * @param longitude
068             * @param latitudeDelta
069             * @param longitudeDelta
070             */
071            public void center(double latitude, double longitude, double latitudeDelta,
072                            double longitudeDelta) {
073                    this.south = latitude - latitudeDelta / 2.0;
074                    this.west = longitude - longitudeDelta / 2.0;
075                    this.north = latitude + latitudeDelta / 2.0;
076                    this.east = longitude + longitudeDelta / 2.0;
077            }
078    
079            /**
080             * Return the latitude of the center of this RectangularLocation.
081             * 
082             * @return
083             */
084            public double getLatitudeCenter() {
085                    return (this.south + this.north) / 2.0;
086            }
087    
088            /**
089             * Return the longitude of the center of this RectangularLocation.
090             * 
091             * @return
092             */
093            public double getLongitudeCenter() {
094                    return (this.west + this.east) / 2.0;
095            }
096    
097            /**
098             * Get the height of the bounding box of this RectangularLocation.
099             * 
100             * @return
101             */
102            public double getLatitudeDelta() {
103                    return Math.abs(this.south - this.north);
104            }
105    
106            /**
107             * Get the width of the bounding box of this RectangularLocation.
108             * 
109             * @return
110             */
111            public double getLongitudeDelta() {
112                    return Math.abs(this.west - this.east);
113            }
114            
115            /**
116             * Returns a CoordinateLocation specifying the southwest corner of this bounding box.
117             * @return
118             */
119            public CoordinateLocation getSouthWest() {
120                    return new CoordinateLocation(getSouth(), getWest());
121            }
122            
123            /**
124             * Returns a CoordinateLocation specifying the northeast corner of this bounding box.
125             * @return
126             */
127            public CoordinateLocation getNorthEast() {
128                    return new CoordinateLocation(getNorth(), getEast());
129            }
130    
131    
132            /**
133             * Returns a CoordinateLocation specifying the center of this bounding box.
134             * 
135             * @return
136             */
137            public CoordinateLocation getCenter() {
138                    return new CoordinateLocation(getLatitudeCenter(), getLongitudeCenter());
139            }
140    
141            /**
142             * @return The southern side of the bounding box (the latitude of the bottom left).
143             */
144            public double getSouth() {
145                    return south;
146            }
147    
148            /**
149             * Sets the southern side of the bounding box (the latitude of the bottom left).
150             * @param south
151             */
152            public void setSouth(double south) {
153                    this.south = south;
154            }
155    
156            /**
157             * @return The western side of the bounding box (the longitude of the bottom-left).
158             */
159            public double getWest() {
160                    return west;
161            }
162    
163            /**
164             * Sets the western side of the bounding box (the longitude of the bottom-left).
165             * @param west
166             */
167            public void setWest(double west) {
168                    this.west = west;
169            }
170    
171            /**
172             * @return The northern side of the bounding box (the latitude of the top-right).
173             */
174            public double getNorth() {
175                    return north;
176            }
177    
178            /**
179             * Sets the northern side of the bounding box (the latitude of the top-right).
180             * @param north
181             */
182            public void setNorth(double north) {
183                    this.north = north;
184            }
185    
186            /**
187             * @return The eastern side of the bounding box (the longitude of the top-right).
188             */
189            public double getEast() {
190                    return east;
191            }
192    
193            /**
194             * Sets the eastern side of the bounding box (the longitude of the top-right).
195             * @param east
196             *            the east to set
197             */
198            public void setEast(double east) {
199                    this.east = east;
200            }
201    
202            /**
203             * <b>This is an internal method. Please ignore it.</b><br><br>
204             * Creates a new RectangularLocation from a dictionary of key-value pairs.
205             * 
206             * @param region
207             * @return
208             */
209            public static RectangularLocation fromCenter(HashMap<String, Object> region) {
210                    @SuppressWarnings("unchecked")
211                    HashMap<String, Object> center = (HashMap<String, Object>) region
212                                    .get("center");
213                    @SuppressWarnings("unchecked")
214                    HashMap<String, Object> span = (HashMap<String, Object>) region
215                                    .get("span");
216                    double latitude = Double.parseDouble(center.get("latitude").toString());
217                    double longitude = Double.parseDouble(center.get("longitude").toString());
218                    double latitudeDelta = Double.parseDouble(span.get("latitude_delta").toString());
219                    double longitudeDelta = Double.parseDouble(span.get("longitude_delta").toString());
220                    RectangularLocation newLocation = new RectangularLocation(0, 0, 0, 0);
221                    newLocation.center(latitude, longitude, latitudeDelta, longitudeDelta);
222                    return newLocation;
223            }
224    
225    }
