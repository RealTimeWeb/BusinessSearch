001    package realtimeweb.businesssearch.main;
002    
003    import java.util.Map;
004    
005    /**
006     * An interface representing a Location. The different locations possible are
007     * listed below, from least specific to most.
008     * 
009     * <ul>
010     * <li>{@link NamedLocation} - Represented by a single string (e.g.
011     * "Blacksburg, VA")</li>
012     * <li>{@link CoordinateLocation} - Represented by a single lat/long (e.g. 42,
013     * 45)</li>
014     * <li>{@link RectangularLocation} - Represented by a pair of lat/longs (e.g.
015     * 42, 45 to 33, 37)</li>
016     * <li>{@link realtimeweb.businesssearch.regular.DetailedLocation
017     * DetailedLocation} - Represented by a coordinate and a postal address</li>
018     * </ul>
019     * 
020     * @author acbart
021     * 
022     */
023    public interface Location {
024    
025            /**
026             * <b> This is an internal method. Please ignore it.</b><br><br>
027             * 
028             * A method for converting the Location to a representation that can be used
029             * to construct a query.
030             * 
031             * @return A mapping between Yelp arguments and the values of the location.
032             */
033            Map<String, String> getFilter();
034    
035    }
